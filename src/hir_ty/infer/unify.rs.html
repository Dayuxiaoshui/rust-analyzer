54    pub(super) fn clauses_for_self_ty(
55        &mut self,
56        self_ty: InferenceVar,
57    ) -> SmallVec<[WhereClause; 4]> {
58        self.table.resolve_obligations_as_possible();
59
60        let root = InferenceVar::from_vid(self.table.infer_ctxt.root_var(self_ty.to_vid()));
61        let pending_obligations = mem::take(&mut self.table.pending_obligations);
62        let obligations = pending_obligations
63            .iter()
64            .filter_map(|obligation| match obligation.to_chalk(self.table.interner).goal.data(Interner) {
65                GoalData::DomainGoal(DomainGoal::Holds(clause)) => {
66                    let ty = match clause {
67                        WhereClause::AliasEq(AliasEq {
68                            alias: AliasTy::Projection(projection),
69                            ..
70                        }) => projection.self_type_parameter(self.db),
71                        WhereClause::Implemented(trait_ref) => {
72                            trait_ref.self_type_parameter(Interner)
73                        }
74                        WhereClause::TypeOutlives(to) => to.ty.clone(),
75                        _ => return None,
76                    };
77                    let ty = self.resolve_ty_shallow(&ty);
78                    if matches!(ty.kind(Interner), TyKind::InferenceVar(iv, TyVariableKind::General) if *iv == root) {
79                        Some(clause.clone())
80                    } else {
81                        None
82                    }
83                }
84                _ => None,
85            })
86            .collect();
87        self.table.pending_obligations = pending_obligations;
88
89        obligations
90    }

93/// Check if types unify.
94///
95/// Note that we consider placeholder types to unify with everything.
96/// This means that there may be some unresolved goals that actually set bounds for the placeholder
97/// type for the types to unify. For example `Option<T>` and `Option<U>` unify although there is
98/// unresolved goal `T = U`.
99pub fn could_unify(
100    db: &dyn HirDatabase,
101    env: Arc<TraitEnvironment>,
102    tys: &Canonical<(Ty, Ty)>,
103) -> bool {
104    unify(db, env, tys).is_some()
105}
106
107/// Check if types unify eagerly making sure there are no unresolved goals.
108///
109/// This means that placeholder types are not considered to unify if there are any bounds set on
110/// them. For example `Option<T>` and `Option<U>` do not unify as we cannot show that `T = U`
111pub fn could_unify_deeply(
112    db: &dyn HirDatabase,
113    env: Arc<TraitEnvironment>,
114    tys: &Canonical<(Ty, Ty)>,
115) -> bool {
116    let mut table = InferenceTable::new(db, env);
117    let vars = make_substitutions(tys, &mut table);
118    let ty1_with_vars = vars.apply(tys.value.0.clone(), Interner);
119    let ty2_with_vars = vars.apply(tys.value.1.clone(), Interner);
120    let ty1_with_vars = table.normalize_associated_types_in(ty1_with_vars);
121    let ty2_with_vars = table.normalize_associated_types_in(ty2_with_vars);
122    table.resolve_obligations_as_possible();
123    table.propagate_diverging_flag();
124    let ty1_with_vars = table.resolve_completely(ty1_with_vars);
125    let ty2_with_vars = table.resolve_completely(ty2_with_vars);
126    table.unify_deeply(&ty1_with_vars, &ty2_with_vars)
127}

129pub(crate) fn unify(
130    db: &dyn HirDatabase,
131    env: Arc<TraitEnvironment>,
132    tys: &Canonical<(Ty, Ty)>,
133) -> Option<Substitution> {
134    let mut table = InferenceTable::new(db, env);
135    let vars = make_substitutions(tys, &mut table);
136    let ty1_with_vars = vars.apply(tys.value.0.clone(), Interner);
137    let ty2_with_vars = vars.apply(tys.value.1.clone(), Interner);
138    if !table.unify(&ty1_with_vars, &ty2_with_vars) {
139        return None;
140    }
141    // default any type vars that weren't unified back to their original bound vars
142    // (kind of hacky)
143    let find_var = |iv| {
144        vars.iter(Interner).position(|v| match v.data(Interner) {
145            GenericArgData::Ty(ty) => ty.inference_var(Interner),
146            GenericArgData::Lifetime(lt) => lt.inference_var(Interner),
147            GenericArgData::Const(c) => c.inference_var(Interner),
148        } == Some(iv))
149    };
150    let fallback = |iv, kind, default, binder| match kind {
151        chalk_ir::VariableKind::Ty(_ty_kind) => find_var(iv)
152            .map_or(default, |i| BoundVar::new(binder, i).to_ty(Interner).cast(Interner)),
153        chalk_ir::VariableKind::Lifetime => find_var(iv)
154            .map_or(default, |i| BoundVar::new(binder, i).to_lifetime(Interner).cast(Interner)),
155        chalk_ir::VariableKind::Const(ty) => find_var(iv)
156            .map_or(default, |i| BoundVar::new(binder, i).to_const(Interner, ty).cast(Interner)),
157    };
158    Some(Substitution::from_iter(
159        Interner,
160        vars.iter(Interner).map(|v| table.resolve_with_fallback(v.clone(), &fallback)),
161    ))
162}

164fn make_substitutions(
165    tys: &chalk_ir::Canonical<(chalk_ir::Ty<Interner>, chalk_ir::Ty<Interner>)>,
166    table: &mut InferenceTable<'_>,
167) -> chalk_ir::Substitution<Interner> {
168    Substitution::from_iter(
169        Interner,
170        tys.binders.iter(Interner).map(|it| match &it.kind {
171            chalk_ir::VariableKind::Ty(_) => table.new_type_var().cast(Interner),
172            // FIXME: maybe wrong?
173            chalk_ir::VariableKind::Lifetime => table.new_type_var().cast(Interner),
174            chalk_ir::VariableKind::Const(ty) => table.new_const_var(ty.clone()).cast(Interner),
175        }),
176    )
177}

221    /// Chalk doesn't know about the `diverging` flag, so when it unifies two
222    /// type variables of which one is diverging, the chosen root might not be
223    /// diverging and we have no way of marking it as such at that time. This
224    /// function goes through all type variables and make sure their root is
225    /// marked as diverging if necessary, so that resolving them gives the right
226    /// result.
227    pub(super) fn propagate_diverging_flag(&mut self) {
228        let mut new_tys = FxHashSet::default();
229        for ty in self.diverging_tys.iter() {
230            match ty.kind(Interner) {
231                TyKind::InferenceVar(var, kind) => match kind {
232                    TyVariableKind::General => {
233                        let root = InferenceVar::from(
234                            self.infer_ctxt.root_var(TyVid::from_u32(var.index())).as_u32(),
235                        );
236                        if root.index() != var.index() {
237                            new_tys.insert(TyKind::InferenceVar(root, *kind).intern(Interner));
238                        }
239                    }
240                    TyVariableKind::Integer => {
241                        let root = InferenceVar::from(
242                            self.infer_ctxt
243                                .inner
244                                .borrow_mut()
245                                .int_unification_table()
246                                .find(IntVid::from_usize(var.index() as usize))
247                                .as_u32(),
248                        );
249                        if root.index() != var.index() {
250                            new_tys.insert(TyKind::InferenceVar(root, *kind).intern(Interner));
251                        }
252                    }
253                    TyVariableKind::Float => {
254                        let root = InferenceVar::from(
255                            self.infer_ctxt
256                                .inner
257                                .borrow_mut()
258                                .float_unification_table()
259                                .find(FloatVid::from_usize(var.index() as usize))
260                                .as_u32(),
261                        );
262                        if root.index() != var.index() {
263                            new_tys.insert(TyKind::InferenceVar(root, *kind).intern(Interner));
264                        }
265                    }
266                },
267                _ => {}
268            }
269        }
270        self.diverging_tys.extend(new_tys);
271    }

277    fn fallback_value(&self, iv: InferenceVar, kind: TyVariableKind) -> Ty {
278        let is_diverging =
279            self.diverging_tys.contains(&TyKind::InferenceVar(iv, kind).intern(Interner));
280        if is_diverging {
281            return TyKind::Never.intern(Interner);
282        }
283        match kind {
284            TyVariableKind::General => TyKind::Error,
285            TyVariableKind::Integer => TyKind::Scalar(Scalar::Int(IntTy::I32)),
286            TyVariableKind::Float => TyKind::Scalar(Scalar::Float(FloatTy::F64)),
287        }
288        .intern(Interner)
289    }

301    /// Recurses through the given type, normalizing associated types mentioned
302    /// in it by replacing them by type variables and registering obligations to
303    /// resolve later. This should be done once for every type we get from some
304    /// type annotation (e.g. from a let type annotation, field type or function
305    /// call). `make_ty` handles this already, but e.g. for field types we need
306    /// to do it as well.
307    pub(crate) fn normalize_associated_types_in<T, U>(&mut self, ty: T) -> T
308    where
309        T: ChalkToNextSolver<'a, U>,
310        U: NextSolverToChalk<'a, T> + rustc_type_ir::TypeFoldable<DbInterner<'a>>,
311    {
312        self.normalize_associated_types_in_ns(ty.to_nextsolver(self.interner))
313            .to_chalk(self.interner)
314    }

318        T: rustc_type_ir::TypeFoldable<DbInterner<'a>>,

324    /// Works almost same as [`Self::normalize_associated_types_in`], but this also resolves shallow
325    /// the inference variables
326    pub(crate) fn eagerly_normalize_and_resolve_shallow_in<T>(&mut self, ty: T) -> T
327    where
328        T: HasInterner<Interner = Interner> + TypeFoldable<Interner>,
329    {
330        fn eagerly_resolve_ty<const N: usize>(
331            table: &mut InferenceTable<'_>,
332            ty: Ty,
333            mut tys: SmallVec<[Ty; N]>,
334        ) -> Ty {
335            if tys.contains(&ty) {
336                return ty;
337            }
338            tys.push(ty.clone());
339
340            match ty.kind(Interner) {
341                TyKind::Alias(AliasTy::Projection(proj_ty)) => {
342                    let ty = table.normalize_projection_ty(proj_ty.clone());
343                    eagerly_resolve_ty(table, ty, tys)
344                }
345                TyKind::InferenceVar(..) => {
346                    let ty = table.resolve_ty_shallow(&ty);
347                    eagerly_resolve_ty(table, ty, tys)
348                }
349                _ => ty,
350            }
351        }
352
353        fold_tys_and_consts(
354            ty,
355            |e, _| match e {
356                Either::Left(ty) => {
357                    Either::Left(eagerly_resolve_ty::<8>(self, ty, SmallVec::new()))
358                }
359                Either::Right(c) => Either::Right(match &c.data(Interner).value {
360                    chalk_ir::ConstValue::Concrete(cc) => match &cc.interned {
361                        crate::ConstScalar::UnevaluatedConst(c_id, subst) => {
362                            // FIXME: same as `normalize_associated_types_in`
363                            if subst.len(Interner) == 0 {
364                                if let Ok(eval) = self.db.const_eval(*c_id, subst.clone(), None) {
365                                    eval
366                                } else {
367                                    unknown_const(c.data(Interner).ty.clone())
368                                }
369                            } else {
370                                unknown_const(c.data(Interner).ty.clone())
371                            }
372                        }
373                        _ => c,
374                    },
375                    _ => c,
376                }),
377            },
378            DebruijnIndex::INNERMOST,
379        )
380    }

390        &mut self,

406    fn new_var(&mut self, kind: TyVariableKind, diverging: bool) -> Ty {
407        let var = match kind {
408            TyVariableKind::General => {
409                let var = self.infer_ctxt.next_ty_vid();
410                InferenceVar::from(var.as_u32())
411            }
412            TyVariableKind::Integer => {
413                let var = self.infer_ctxt.next_int_vid();
414                InferenceVar::from(var.as_u32())
415            }
416            TyVariableKind::Float => {
417                let var = self.infer_ctxt.next_float_vid();
418                InferenceVar::from(var.as_u32())
419            }
420        };
421
422        let ty = var.to_ty(Interner, kind);
423        if diverging {
424            self.diverging_tys.insert(ty.clone());
425        }
426        ty
427    }

514    pub(crate) fn resolve_completely<T, U>(&mut self, t: T) -> T
515    where
516        T: HasInterner<Interner = Interner> + TypeFoldable<Interner> + ChalkToNextSolver<'a, U>,
517        U: NextSolverToChalk<'a, T> + rustc_type_ir::TypeFoldable<DbInterner<'a>>,
518    {
519        let t = self.resolve_with_fallback(t, &|_, _, d, _| d);
520        let t = self.normalize_associated_types_in(t);
521        // let t = self.resolve_opaque_tys_in(t);
522        // Resolve again, because maybe normalization inserted infer vars.
523        self.resolve_with_fallback(t, &|_, _, d, _| d)
524    }

526    /// Apply a fallback to unresolved scalar types. Integer type variables and float type
527    /// variables are replaced with i32 and f64, respectively.
528    ///
529    /// This method is only intended to be called just before returning inference results (i.e. in
530    /// `InferenceContext::resolve_all()`).
531    ///
532    /// FIXME: This method currently doesn't apply fallback to unconstrained general type variables
533    /// whereas rustc replaces them with `()` or `!`.
534    pub(super) fn fallback_if_possible(&mut self) {
535        let int_fallback = TyKind::Scalar(Scalar::Int(IntTy::I32)).intern(Interner);
536        let float_fallback = TyKind::Scalar(Scalar::Float(FloatTy::F64)).intern(Interner);
537
538        let int_vars = self.infer_ctxt.inner.borrow_mut().int_unification_table().len();
539        for v in 0..int_vars {
540            let var = InferenceVar::from(v as u32).to_ty(Interner, TyVariableKind::Integer);
541            let maybe_resolved = self.resolve_ty_shallow(&var);
542            if let TyKind::InferenceVar(_, kind) = maybe_resolved.kind(Interner) {
543                // I don't think we can ever unify these vars with float vars, but keep this here for now
544                let fallback = match kind {
545                    TyVariableKind::Integer => &int_fallback,
546                    TyVariableKind::Float => &float_fallback,
547                    TyVariableKind::General => unreachable!(),
548                };
549                self.unify(&var, fallback);
550            }
551        }
552        let float_vars = self.infer_ctxt.inner.borrow_mut().float_unification_table().len();
553        for v in 0..float_vars {
554            let var = InferenceVar::from(v as u32).to_ty(Interner, TyVariableKind::Float);
555            let maybe_resolved = self.resolve_ty_shallow(&var);
556            if let TyKind::InferenceVar(_, kind) = maybe_resolved.kind(Interner) {
557                // I don't think we can ever unify these vars with float vars, but keep this here for now
558                let fallback = match kind {
559                    TyVariableKind::Integer => &int_fallback,
560                    TyVariableKind::Float => &float_fallback,
561                    TyVariableKind::General => unreachable!(),
562                };
563                self.unify(&var, fallback);
564            }
565        }
566    }

568    /// Unify two relatable values (e.g. `Ty`) and register new trait goals that arise from that.
569    pub(crate) fn unify<T: ChalkToNextSolver<'a, U>, U: Relate<DbInterner<'a>>>(
570        &mut self,
571        ty1: &T,
572        ty2: &T,
573    ) -> bool {
574        let result = match self.try_unify(ty1, ty2) {
575            Ok(r) => r,
576            Err(_) => return false,
577        };
578        self.register_infer_ok(result);
579        true
580    }

582    /// Unify two relatable values (e.g. `Ty`) and check whether trait goals which arise from that could be fulfilled
583    pub(crate) fn unify_deeply<T: ChalkToNextSolver<'a, U>, U: Relate<DbInterner<'a>>>(
584        &mut self,
585        ty1: &T,
586        ty2: &T,
587    ) -> bool {
588        let result = match self.try_unify(ty1, ty2) {
589            Ok(r) => r,
590            Err(_) => return false,
591        };
592        result.goals.into_iter().all(|goal| {
593            matches!(next_trait_solve_in_ctxt(&self.infer_ctxt, goal), Ok((_, Certainty::Yes)))
594        })
595    }

597    /// Unify two relatable values (e.g. `Ty`) and return new trait goals arising from it, so the
598    /// caller needs to deal with them.
599    pub(crate) fn try_unify<T: ChalkToNextSolver<'a, U>, U: Relate<DbInterner<'a>>>(
600        &mut self,
601        t1: &T,
602        t2: &T,
603    ) -> InferResult<'a, ()> {
604        let param_env = self.trait_env.env.to_nextsolver(self.interner);
605        let lhs = t1.to_nextsolver(self.interner);
606        let rhs = t2.to_nextsolver(self.interner);
607        let variance = rustc_type_ir::Variance::Invariant;
608        let span = crate::next_solver::Span::dummy();
609        match self.infer_ctxt.relate(param_env, lhs, variance, rhs, span) {
610            Ok(goals) => Ok(InferOk { goals, value: () }),
611            Err(_) => Err(TypeError),
612        }
613    }

615    /// If `ty` is a type variable with known type, returns that type;
616    /// otherwise, return ty.
617    #[tracing::instrument(skip(self))]
618    pub(crate) fn resolve_ty_shallow(&mut self, ty: &Ty) -> Ty {
619        if !ty.data(Interner).flags.intersects(chalk_ir::TypeFlags::HAS_FREE_LOCAL_NAMES) {
620            return ty.clone();
621        }
622        self.infer_ctxt
623            .resolve_vars_if_possible(ty.to_nextsolver(self.interner))
624            .to_chalk(self.interner)

627    pub(crate) fn resolve_vars_with_obligations<T>(&mut self, t: T) -> T
628    where
629        T: rustc_type_ir::TypeFoldable<DbInterner<'a>>,
630    {
631        use rustc_type_ir::TypeVisitableExt;
632
633        if !t.has_non_region_infer() {
634            return t;
635        }
636
637        let t = self.infer_ctxt.resolve_vars_if_possible(t);
638
639        if !t.has_non_region_infer() {
640            return t;
641        }
642
643        self.resolve_obligations_as_possible();
644        self.infer_ctxt.resolve_vars_if_possible(t)
645    }

647    pub(crate) fn structurally_resolve_type(&mut self, ty: &Ty) -> Ty {
648        if let TyKind::Alias(..) = ty.kind(Interner) {
649            self.structurally_normalize_ty(ty)
650        } else {
651            self.resolve_vars_with_obligations(ty.to_nextsolver(self.interner))
652                .to_chalk(self.interner)
653        }
654    }

656    fn structurally_normalize_ty(&mut self, ty: &Ty) -> Ty {
657        self.structurally_normalize_term(ty.to_nextsolver(self.interner).into())
658            .expect_ty()
659            .to_chalk(self.interner)
660    }

662    fn structurally_normalize_term(&mut self, term: Term<'a>) -> Term<'a> {
663        if term.to_alias_term().is_none() {
664            return term;
665        }
666
667        let new_infer = self.infer_ctxt.next_term_var_of_kind(term);
668
669        self.register_obligation(Predicate::new(
670            self.interner,
671            Binder::dummy(PredicateKind::AliasRelate(
672                term,
673                new_infer,
674                AliasRelationDirection::Equate,
675            )),
676        ));
677        self.resolve_obligations_as_possible();
678        let res = self.infer_ctxt.resolve_vars_if_possible(new_infer);
679        if res == new_infer { term } else { res }
680    }

719    /// Checks an obligation without registering it. Useful mostly to check
720    /// whether a trait *might* be implemented before deciding to 'lock in' the
721    /// choice (during e.g. method resolution or deref).
722    #[tracing::instrument(level = "debug", skip(self))]
723    pub(crate) fn try_obligation(&mut self, goal: Goal) -> NextTraitSolveResult {
724        let in_env = InEnvironment::new(&self.trait_env.env, goal);
725        let canonicalized = self.canonicalize(in_env.to_nextsolver(self.interner));
726
727        next_trait_solve_canonical_in_ctxt(&self.infer_ctxt, canonicalized)
728    }

730    #[tracing::instrument(level = "debug", skip(self))]
731    pub(crate) fn solve_obligation(&mut self, goal: Goal) -> Result<Certainty, NoSolution> {
732        let goal = InEnvironment::new(&self.trait_env.env, goal);
733        let goal = goal.to_nextsolver(self.interner);
734        let result = next_trait_solve_in_ctxt(&self.infer_ctxt, goal);
735        result.map(|m| m.1)
736    }

738    pub(crate) fn register_obligation(&mut self, predicate: Predicate<'a>) {
739        let goal = next_solver::Goal {
740            param_env: self.trait_env.env.to_nextsolver(self.interner),
741            predicate,
742        };
743        self.register_obligation_in_env(goal)
744    }
745
746    #[tracing::instrument(level = "debug", skip(self))]
747    fn register_obligation_in_env(
748        &mut self,
749        goal: next_solver::Goal<'a, next_solver::Predicate<'a>>,
750    ) {
751        let result = next_trait_solve_in_ctxt(&self.infer_ctxt, goal);
752        tracing::debug!(?result);
753        match result {
754            Ok((_, Certainty::Yes)) => {}
755            Err(rustc_type_ir::solve::NoSolution) => {}
756            Ok((_, Certainty::Maybe(_))) => {
757                self.pending_obligations.push(goal);
758            }
759        }
760    }
761
762    pub(crate) fn register_infer_ok<T>(&mut self, infer_ok: InferOk<'a, T>) {
763        infer_ok.goals.into_iter().for_each(|goal| self.register_obligation_in_env(goal));
764    }

766    pub(crate) fn resolve_obligations_as_possible(&mut self) {
767        let _span = tracing::info_span!("resolve_obligations_as_possible").entered();
768        let mut changed = true;
769        while mem::take(&mut changed) {
770            let mut obligations = mem::take(&mut self.pending_obligations);
771
772            for goal in obligations.drain(..) {
773                tracing::debug!(obligation = ?goal);
774
775                let result = next_trait_solve_in_ctxt(&self.infer_ctxt, goal);
776                let (has_changed, certainty) = match result {
777                    Ok(result) => result,
778                    Err(_) => {
779                        continue;
780                    }
781                };
782
783                if matches!(has_changed, HasChanged::Yes) {
784                    changed = true;
785                }
786
787                match certainty {
788                    Certainty::Yes => {}
789                    Certainty::Maybe(_) => self.pending_obligations.push(goal),
790                }
791            }
792        }
793    }

795    pub(crate) fn fudge_inference<T: TypeFoldable<Interner>>(
796        &mut self,
797        f: impl FnOnce(&mut Self) -> T,
798    ) -> T {
799        use chalk_ir::fold::TypeFolder;
800
801        #[derive(chalk_derive::FallibleTypeFolder)]
802        #[has_interner(Interner)]
803        struct VarFudger<'a, 'b> {
804            table: &'a mut InferenceTable<'b>,
805            highest_known_var: InferenceVar,
806        }
807        impl TypeFolder<Interner> for VarFudger<'_, '_> {
808            fn as_dyn(&mut self) -> &mut dyn TypeFolder<Interner> {
809                self
810            }
811
812            fn interner(&self) -> Interner {
813                Interner
814            }
815
816            fn fold_inference_ty(
817                &mut self,
818                var: chalk_ir::InferenceVar,
819                kind: TyVariableKind,
820                _outer_binder: chalk_ir::DebruijnIndex,
821            ) -> chalk_ir::Ty<Interner> {
822                if var < self.highest_known_var {
823                    var.to_ty(Interner, kind)
824                } else {
825                    self.table.new_type_var()
826                }
827            }
828
829            fn fold_inference_lifetime(
830                &mut self,
831                var: chalk_ir::InferenceVar,
832                _outer_binder: chalk_ir::DebruijnIndex,
833            ) -> chalk_ir::Lifetime<Interner> {
834                if var < self.highest_known_var {
835                    var.to_lifetime(Interner)
836                } else {
837                    self.table.new_lifetime_var()
838                }
839            }
840
841            fn fold_inference_const(
842                &mut self,
843                ty: chalk_ir::Ty<Interner>,
844                var: chalk_ir::InferenceVar,
845                _outer_binder: chalk_ir::DebruijnIndex,
846            ) -> chalk_ir::Const<Interner> {
847                if var < self.highest_known_var {
848                    var.to_const(Interner, ty)
849                } else {
850                    self.table.new_const_var(ty)
851                }
852            }
853        }
854
855        let snapshot = self.snapshot();
856        let highest_known_var = self.new_type_var().inference_var(Interner).expect("inference_var");
857        let result = f(self);
858        self.rollback_to(snapshot);
859        result
860            .fold_with(&mut VarFudger { table: self, highest_known_var }, DebruijnIndex::INNERMOST)
861    }

863    pub(crate) fn callable_sig(
864        &mut self,
865        ty: &Ty,
866        num_args: usize,
867    ) -> Option<(Option<FnTrait>, Vec<Ty>, Ty)> {
868        match ty.callable_sig(self.db) {
869            Some(sig) => Some((None, sig.params().to_vec(), sig.ret().clone())),
870            None => {
871                let (f, args_ty, return_ty) = self.callable_sig_from_fn_trait(ty, num_args)?;
872                Some((Some(f), args_ty, return_ty))
873            }
874        }
875    }
876
877    fn callable_sig_from_fn_trait(
878        &mut self,
879        ty: &Ty,
880        num_args: usize,
881    ) -> Option<(FnTrait, Vec<Ty>, Ty)> {
882        for (fn_trait_name, output_assoc_name, subtraits) in [
883            (FnTrait::FnOnce, sym::Output, &[FnTrait::Fn, FnTrait::FnMut][..]),
884            (FnTrait::AsyncFnMut, sym::CallRefFuture, &[FnTrait::AsyncFn]),
885            (FnTrait::AsyncFnOnce, sym::CallOnceFuture, &[]),
886        ] {
887            let krate = self.trait_env.krate;
888            let fn_trait = fn_trait_name.get_id(self.db, krate)?;
889            let trait_data = fn_trait.trait_items(self.db);
890            let output_assoc_type =
891                trait_data.associated_type_by_name(&Name::new_symbol_root(output_assoc_name))?;
892
893            let mut arg_tys = Vec::with_capacity(num_args);
894            let arg_ty = TyBuilder::tuple(num_args)
895                .fill(|it| {
896                    let arg = match it {
897                        ParamKind::Type => self.new_type_var(),
898                        ParamKind::Lifetime => unreachable!("Tuple with lifetime parameter"),
899                        ParamKind::Const(_) => unreachable!("Tuple with const parameter"),
900                    };
901                    arg_tys.push(arg.clone());
902                    arg.cast(Interner)
903                })
904                .build();
905
906            let b = TyBuilder::trait_ref(self.db, fn_trait);
907            if b.remaining() != 2 {
908                return None;
909            }
910            let mut trait_ref = b.push(ty.clone()).push(arg_ty).build();
911
912            let projection = TyBuilder::assoc_type_projection(
913                self.db,
914                output_assoc_type,
915                Some(trait_ref.substitution.clone()),
916            )
917            .fill_with_unknown()
918            .build();
919
920            let goal: Goal = trait_ref.clone().cast(Interner);
921            if !self.try_obligation(goal.clone()).no_solution() {
922                self.register_obligation(goal.to_nextsolver(self.interner));
923                let return_ty = self.normalize_projection_ty(projection);
924                for &fn_x in subtraits {
925                    let fn_x_trait = fn_x.get_id(self.db, krate)?;
926                    trait_ref.trait_id = to_chalk_trait_id(fn_x_trait);
927                    let goal = trait_ref.clone().cast(Interner);
928                    if !self.try_obligation(goal).no_solution() {
929                        return Some((fn_x, arg_tys, return_ty));
930                    }
931                }
932                return Some((fn_trait_name, arg_tys, return_ty));
933            }
934        }
935        None
936    }

993    /// Replaces ConstScalar::Unknown by a new type var, so we can maybe still infer it.
994    pub(super) fn insert_const_vars_shallow(&mut self, c: Const) -> Const {
995        let data = c.data(Interner);
996        match &data.value {
997            ConstValue::Concrete(cc) => match &cc.interned {
998                crate::ConstScalar::Unknown => self.new_const_var(data.ty.clone()),
999                // try to evaluate unevaluated const. Replace with new var if const eval failed.
1000                crate::ConstScalar::UnevaluatedConst(id, subst) => {
1001                    if let Ok(eval) = self.db.const_eval(*id, subst.clone(), None) {
1002                        eval
1003                    } else {
1004                        self.new_const_var(data.ty.clone())
1005                    }
1006                }
1007                _ => c,
1008            },
1009            _ => c,
1010        }

1013    /// Check if given type is `Sized` or not
1014    pub(crate) fn is_sized(&mut self, ty: &Ty) -> bool {
1015        fn short_circuit_trivial_tys(ty: &Ty) -> Option<bool> {
1016            match ty.kind(Interner) {
1017                TyKind::Scalar(..)
1018                | TyKind::Ref(..)
1019                | TyKind::Raw(..)
1020                | TyKind::Never
1021                | TyKind::FnDef(..)
1022                | TyKind::Array(..)
1023                | TyKind::Function(..) => Some(true),
1024                TyKind::Slice(..) | TyKind::Str | TyKind::Dyn(..) => Some(false),
1025                _ => None,
1026            }
1027        }
1028
1029        let mut ty = ty.clone();
1030        ty = self.eagerly_normalize_and_resolve_shallow_in(ty);
1031        if let Some(sized) = short_circuit_trivial_tys(&ty) {
1032            return sized;
1033        }
1034
1035        {
1036            let mut structs = SmallVec::<[_; 8]>::new();
1037            // Must use a loop here and not recursion because otherwise users will conduct completely
1038            // artificial examples of structs that have themselves as the tail field and complain r-a crashes.
1039            while let Some((AdtId::StructId(id), subst)) = ty.as_adt() {
1040                let struct_data = id.fields(self.db);
1041                if let Some((last_field, _)) = struct_data.fields().iter().next_back() {
1042                    let last_field_ty = self.db.field_types(id.into())[last_field]
1043                        .clone()
1044                        .substitute(Interner, subst);
1045                    if structs.contains(&ty) {
1046                        // A struct recursively contains itself as a tail field somewhere.
1047                        return true; // Don't overload the users with too many errors.
1048                    }
1049                    structs.push(ty);
1050                    // Structs can have DST as its last field and such cases are not handled
1051                    // as unsized by the chalk, so we do this manually.
1052                    ty = last_field_ty;
1053                    ty = self.eagerly_normalize_and_resolve_shallow_in(ty);
1054                    if let Some(sized) = short_circuit_trivial_tys(&ty) {
1055                        return sized;
1056                    }
1057                } else {
1058                    break;
1059                };
1060            }
1061        }
1062
1063        let Some(sized) = LangItem::Sized.resolve_trait(self.db, self.trait_env.krate) else {
1064            return false;
1065        };
1066        let sized_pred = WhereClause::Implemented(TraitRef {
1067            trait_id: to_chalk_trait_id(sized),
1068            substitution: Substitution::from1(Interner, ty),
1069        });
1070        let goal = GoalData::DomainGoal(chalk_ir::DomainGoal::Holds(sized_pred)).intern(Interner);
1071        self.try_obligation(goal).certain()
1072    }
1073}

1129            match kind {
1130                TyVariableKind::General => {
1131                    let vid = self.table.infer_ctxt.root_var(TyVid::from(var.index()));
1132                    let var = InferenceVar::from(vid.as_u32());
1133                    if self.var_stack.contains(&(var, VarKind::Ty(kind))) {
1134                        // recursive type
1135                        let default = self.table.fallback_value(var, kind).cast(Interner);
1136                        return (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1137                            .assert_ty_ref(Interner)
1138                            .clone();
1139                    }
1140                    if let Ok(known_ty) = self.table.infer_ctxt.probe_ty_var(vid) {
1141                        let known_ty: Ty = known_ty.to_chalk(self.table.interner);
1142                        // known_ty may contain other variables that are known by now
1143                        self.var_stack.push((var, VarKind::Ty(kind)));
1144                        let result = known_ty.fold_with(self, outer_binder);
1145                        self.var_stack.pop();
1146                        result
1147                    } else {
1148                        let default = self.table.fallback_value(var, kind).cast(Interner);
1149                        (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1150                            .assert_ty_ref(Interner)
1151                            .clone()
1152                    }
1153                }
1154                TyVariableKind::Integer => {
1155                    let vid = self
1156                        .table
1157                        .infer_ctxt
1158                        .inner
1159                        .borrow_mut()
1160                        .int_unification_table()
1161                        .find(IntVid::from(var.index()));
1162                    let var = InferenceVar::from(vid.as_u32());
1163                    if self.var_stack.contains(&(var, VarKind::Ty(kind))) {
1164                        // recursive type
1165                        let default = self.table.fallback_value(var, kind).cast(Interner);
1166                        return (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1167                            .assert_ty_ref(Interner)
1168                            .clone();
1169                    }
1170                    if let Some(known_ty) = self.table.infer_ctxt.resolve_int_var(vid) {
1171                        let known_ty: Ty = known_ty.to_chalk(self.table.interner);
1172                        // known_ty may contain other variables that are known by now
1173                        self.var_stack.push((var, VarKind::Ty(kind)));
1174                        let result = known_ty.fold_with(self, outer_binder);
1175                        self.var_stack.pop();
1176                        result
1177                    } else {
1178                        let default = self.table.fallback_value(var, kind).cast(Interner);
1179                        (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1180                            .assert_ty_ref(Interner)
1181                            .clone()
1182                    }
1183                }
1184                TyVariableKind::Float => {
1185                    let vid = self
1186                        .table
1187                        .infer_ctxt
1188                        .inner
1189                        .borrow_mut()
1190                        .float_unification_table()
1191                        .find(FloatVid::from(var.index()));
1192                    let var = InferenceVar::from(vid.as_u32());
1193                    if self.var_stack.contains(&(var, VarKind::Ty(kind))) {
1194                        // recursive type
1195                        let default = self.table.fallback_value(var, kind).cast(Interner);
1196                        return (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1197                            .assert_ty_ref(Interner)
1198                            .clone();
1199                    }
1200                    if let Some(known_ty) = self.table.infer_ctxt.resolve_float_var(vid) {
1201                        let known_ty: Ty = known_ty.to_chalk(self.table.interner);
1202                        // known_ty may contain other variables that are known by now
1203                        self.var_stack.push((var, VarKind::Ty(kind)));
1204                        let result = known_ty.fold_with(self, outer_binder);
1205                        self.var_stack.pop();
1206                        result
1207                    } else {
1208                        let default = self.table.fallback_value(var, kind).cast(Interner);
1209                        (self.fallback)(var, VariableKind::Ty(kind), default, outer_binder)
1210                            .assert_ty_ref(Interner)
1211                            .clone()
1212                    }
1213                }
1214            }
1215        }
1216
1217        fn fold_inference_const(
1218            &mut self,
1219            ty: Ty,
1220            var: InferenceVar,
1221            outer_binder: DebruijnIndex,
1222        ) -> Const {
1223            let vid = self
1224                .table
1225                .infer_ctxt
1226                .root_const_var(rustc_type_ir::ConstVid::from_u32(var.index()));
1227            let var = InferenceVar::from(vid.as_u32());
1228            let default = ConstData {
1229                ty: ty.clone(),
1230                value: ConstValue::Concrete(ConcreteConst { interned: ConstScalar::Unknown }),
1231            }
1232            .intern(Interner)
1233            .cast(Interner);
1234            if self.var_stack.contains(&(var, VarKind::Const)) {
1235                // recursive
1236                return (self.fallback)(var, VariableKind::Const(ty), default, outer_binder)
1237                    .assert_const_ref(Interner)
1238                    .clone();
1239            }
1240            if let Ok(known_const) = self.table.infer_ctxt.probe_const_var(vid) {
1241                let known_const: Const = known_const.to_chalk(self.table.interner);
1242                // known_ty may contain other variables that are known by now
1243                self.var_stack.push((var, VarKind::Const));
1244                let result = known_const.fold_with(self, outer_binder);
1245                self.var_stack.pop();
1246                result
1247            } else {
1248                (self.fallback)(var, VariableKind::Const(ty), default, outer_binder)
1249                    .assert_const_ref(Interner)
1250                    .clone()
1251            }
1252        }
1253
1254        fn fold_inference_lifetime(
1255            &mut self,
1256            _var: InferenceVar,
1257            _outer_binder: DebruijnIndex,
1258        ) -> Lifetime {
1259            // fall back all lifetimes to 'error -- currently we don't deal
1260            // with any lifetimes, but we can sometimes get some lifetime
1261            // variables through Chalk's unification, and this at least makes
1262            // sure we don't leak them outside of inference
1263            crate::error_lifetime()
1264        }
1265    }
1266}

1268/// This expects its input to be resolved.
1269struct Normalizer<'a, 'b> {
1270    table: &'a mut InferenceTable<'b>,
1271}
1272
1273impl<'db> Normalizer<'_, 'db> {
1274    fn normalize_alias_term(
1275        &mut self,
1276        alias_term: crate::next_solver::Term<'db>,
1277    ) -> crate::next_solver::Term<'db> {
1278        let infer_term = self.table.infer_ctxt.next_term_var_of_kind(alias_term);
1279        let obligation = crate::next_solver::Predicate::new(
1280            self.table.interner,
1281            crate::next_solver::Binder::dummy(crate::next_solver::PredicateKind::AliasRelate(
1282                alias_term,
1283                infer_term,
1284                rustc_type_ir::AliasRelationDirection::Equate,
1285            )),
1286        );
1287        self.table.register_obligation(obligation);
1288        let term = self.table.resolve_vars_with_obligations(infer_term);
1289        // Now normalize the result, because maybe it contains more aliases.
1290        match term {
1291            Term::Ty(term) => term.super_fold_with(self).into(),
1292            Term::Const(term) => term.super_fold_with(self).into(),
1293        }
1294    }
1295}
1296
1297impl<'db> rustc_type_ir::TypeFolder<DbInterner<'db>> for Normalizer<'_, 'db> {
1298    fn cx(&self) -> DbInterner<'db> {
1299        self.table.interner
1300    }
1301
1302    fn fold_ty(&mut self, ty: crate::next_solver::Ty<'db>) -> crate::next_solver::Ty<'db> {
1303        if !ty.has_aliases() {
1304            return ty;
1305        }
1306
1307        let crate::next_solver::TyKind::Alias(..) = ty.kind() else {
1308            return ty.super_fold_with(self);
1309        };
1310        // FIXME: Handle escaping bound vars by replacing them with placeholders (relevant to when we handle HRTB only).
1311        self.normalize_alias_term(ty.into()).expect_type()
1312    }
1313
1314    fn fold_const(&mut self, ct: crate::next_solver::Const<'db>) -> crate::next_solver::Const<'db> {
1315        if !ct.has_aliases() {
1316            return ct;
1317        }
1318
1319        let crate::next_solver::ConstKind::Unevaluated(..) = ct.kind() else {
1320            return ct.super_fold_with(self);
1321        };
1322        // FIXME: Handle escaping bound vars by replacing them with placeholders (relevant to when we handle HRTB only).
1323        self.normalize_alias_term(ct.into()).expect_const()
1324    }
1325}
