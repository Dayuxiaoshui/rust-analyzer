45/// This is used as a key for indexing impls.
46#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
47pub enum TyFingerprint {
48    // These are lang item impls:
49    Str,
50    Slice,
51    Array,
52    Never,
53    Ref(Mutability),
54    RawPtr(Mutability),
55    Scalar(Scalar),
56    // These can have user-defined impls:
57    Adt(hir_def::AdtId),
58    Dyn(TraitId),
59    ForeignType(ForeignDefId),
60    // These only exist for trait impls
61    Unit,
62    Unnameable,
63    Function(u32),
64}
65
66impl TyFingerprint {
67    /// Creates a TyFingerprint for looking up an inherent impl. Only certain
68    /// types can have inherent impls: if we have some `struct S`, we can have
69    /// an `impl S`, but not `impl &S`. Hence, this will return `None` for
70    /// reference types and such.
71    pub fn for_inherent_impl(ty: &Ty) -> Option<TyFingerprint> {
72        let fp = match ty.kind(Interner) {
73            TyKind::Str => TyFingerprint::Str,
74            TyKind::Never => TyFingerprint::Never,
75            TyKind::Slice(..) => TyFingerprint::Slice,
76            TyKind::Array(..) => TyFingerprint::Array,
77            TyKind::Scalar(scalar) => TyFingerprint::Scalar(*scalar),
78            TyKind::Adt(AdtId(adt), _) => TyFingerprint::Adt(*adt),
79            TyKind::Raw(mutability, ..) => TyFingerprint::RawPtr(*mutability),
80            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(*alias_id),
81            TyKind::Dyn(_) => ty.dyn_trait().map(TyFingerprint::Dyn)?,
82            _ => return None,
83        };
84        Some(fp)
85    }

87    /// Creates a TyFingerprint for looking up a trait impl.
88    pub fn for_trait_impl(ty: &Ty) -> Option<TyFingerprint> {
89        let fp = match ty.kind(Interner) {
90            TyKind::Str => TyFingerprint::Str,
91            TyKind::Never => TyFingerprint::Never,
92            TyKind::Slice(..) => TyFingerprint::Slice,
93            TyKind::Array(..) => TyFingerprint::Array,
94            TyKind::Scalar(scalar) => TyFingerprint::Scalar(*scalar),
95            TyKind::Adt(AdtId(adt), _) => TyFingerprint::Adt(*adt),
96            TyKind::Raw(mutability, ..) => TyFingerprint::RawPtr(*mutability),
97            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(*alias_id),
98            TyKind::Dyn(_) => ty.dyn_trait().map(TyFingerprint::Dyn)?,
99            TyKind::Ref(mutability, _, _) => TyFingerprint::Ref(*mutability),
100            TyKind::Tuple(_, subst) => {
101                let first_ty = subst.interned().first().map(|arg| arg.assert_ty_ref(Interner));
102                match first_ty {
103                    Some(ty) => return TyFingerprint::for_trait_impl(ty),
104                    None => TyFingerprint::Unit,
105                }
106            }
107            TyKind::AssociatedType(_, _)
108            // FIXME(next-solver): Putting `Alias` here is *probably* incorrect, AFAIK it should return `None`. But this breaks
109            // flyimport, which uses an incorrect but fast method resolution algorithm. Therefore we put it here,
110            // because this function is only called by flyimport, and anyway we should get rid of `TyFingerprint`
111            // and switch to `rustc_type_ir`'s `SimplifiedType`.
112            | TyKind::Alias(_)
113            | TyKind::OpaqueType(_, _)
114            | TyKind::FnDef(_, _)
115            | TyKind::Closure(_, _)
116            | TyKind::Coroutine(..)
117            | TyKind::CoroutineWitness(..) => TyFingerprint::Unnameable,
118            TyKind::Function(fn_ptr) => {
119                TyFingerprint::Function(fn_ptr.substitution.0.len(Interner) as u32)
120            }
121            TyKind::Placeholder(_)
122            | TyKind::BoundVar(_)
123            | TyKind::InferenceVar(_, _)
124            | TyKind::Error => return None,
125        };
126        Some(fp)
127    }

129    /// Creates a TyFingerprint for looking up a trait impl.
130    pub fn for_trait_impl_ns<'db>(ty: &next_solver::Ty<'db>) -> Option<TyFingerprint> {
131        use rustc_type_ir::TyKind;
132        let fp = match (*ty).kind() {
133            TyKind::Str => TyFingerprint::Str,
134            TyKind::Never => TyFingerprint::Never,
135            TyKind::Slice(..) => TyFingerprint::Slice,
136            TyKind::Array(..) => TyFingerprint::Array,
137            TyKind::Int(int) => TyFingerprint::Scalar(Scalar::Int(match int {
138                rustc_type_ir::IntTy::Isize => IntTy::Isize,
139                rustc_type_ir::IntTy::I8 => IntTy::I8,
140                rustc_type_ir::IntTy::I16 => IntTy::I16,
141                rustc_type_ir::IntTy::I32 => IntTy::I32,
142                rustc_type_ir::IntTy::I64 => IntTy::I64,
143                rustc_type_ir::IntTy::I128 => IntTy::I128,
144            })),
145            TyKind::Uint(uint) => TyFingerprint::Scalar(Scalar::Uint(match uint {
146                rustc_type_ir::UintTy::Usize => UintTy::Usize,
147                rustc_type_ir::UintTy::U8 => UintTy::U8,
148                rustc_type_ir::UintTy::U16 => UintTy::U16,
149                rustc_type_ir::UintTy::U32 => UintTy::U32,
150                rustc_type_ir::UintTy::U64 => UintTy::U64,
151                rustc_type_ir::UintTy::U128 => UintTy::U128,
152            })),
153            TyKind::Float(float) => TyFingerprint::Scalar(Scalar::Float(match float {
154                rustc_type_ir::FloatTy::F16 => FloatTy::F16,
155                rustc_type_ir::FloatTy::F32 => FloatTy::F32,
156                rustc_type_ir::FloatTy::F64 => FloatTy::F64,
157                rustc_type_ir::FloatTy::F128 => FloatTy::F128,
158            })),
159            TyKind::Bool => TyFingerprint::Scalar(Scalar::Bool),
160            TyKind::Char => TyFingerprint::Scalar(Scalar::Char),
161            TyKind::Adt(def, _) => TyFingerprint::Adt(def.inner().id),
162            TyKind::RawPtr(.., mutability) => match mutability {
163                rustc_ast_ir::Mutability::Mut => TyFingerprint::RawPtr(Mutability::Mut),
164                rustc_ast_ir::Mutability::Not => TyFingerprint::RawPtr(Mutability::Not),
165            },
166            TyKind::Foreign(def) => TyFingerprint::ForeignType(crate::to_foreign_def_id(def.0)),
167            TyKind::Dynamic(bounds, _, _) => {
168                let trait_ref = bounds
169                    .as_slice()
170                    .iter()
171                    .map(|b| (*b).skip_binder())
172                    .filter_map(|b| match b {
173                        rustc_type_ir::ExistentialPredicate::Trait(t) => Some(t.def_id),
174                        _ => None,
175                    })
176                    .next()?;
177                TyFingerprint::Dyn(trait_ref.0)
178            }
179            TyKind::Ref(_, _, mutability) => match mutability {
180                rustc_ast_ir::Mutability::Mut => TyFingerprint::Ref(Mutability::Mut),
181                rustc_ast_ir::Mutability::Not => TyFingerprint::Ref(Mutability::Not),
182            },
183            TyKind::Tuple(tys) => {
184                let first_ty = tys.as_slice().iter().next();
185                match first_ty {
186                    Some(ty) => return TyFingerprint::for_trait_impl_ns(ty),
187                    None => TyFingerprint::Unit,
188                }
189            }
190            TyKind::FnDef(_, _)
191            | TyKind::Closure(_, _)
192            | TyKind::Coroutine(..)
193            | TyKind::CoroutineWitness(..)
194            | TyKind::Pat(..)
195            | TyKind::CoroutineClosure(..) => TyFingerprint::Unnameable,
196            TyKind::FnPtr(sig, _) => {
197                TyFingerprint::Function(sig.inputs().skip_binder().len() as u32)
198            }
199            TyKind::Alias(..)
200            | TyKind::Placeholder(_)
201            | TyKind::Bound(..)
202            | TyKind::Infer(_)
203            | TyKind::Error(_)
204            | TyKind::Param(..)
205            | TyKind::UnsafeBinder(..) => return None,
206        };
207        Some(fp)
208    }
209}

236/// Trait impls defined or available in some crate.
237#[derive(Debug, Eq, PartialEq)]
238pub struct TraitImpls {
239    // If the `Option<TyFingerprint>` is `None`, the impl may apply to any self type.
240    map: TraitFpMap,
241}

243impl TraitImpls {
244    pub(crate) fn trait_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
245        let _p = tracing::info_span!("trait_impls_in_crate_query", ?krate).entered();
246        let mut impls = FxHashMap::default();
247
248        Self::collect_def_map(db, &mut impls, crate_def_map(db, krate));
249
250        Arc::new(Self::finish(impls))
251    }
252
253    pub(crate) fn trait_impls_in_block_query(
254        db: &dyn HirDatabase,
255        block: BlockId,
256    ) -> Option<Arc<Self>> {
257        let _p = tracing::info_span!("trait_impls_in_block_query").entered();
258        let mut impls = FxHashMap::default();
259
260        Self::collect_def_map(db, &mut impls, block_def_map(db, block));
261
262        if impls.is_empty() { None } else { Some(Arc::new(Self::finish(impls))) }
263    }
264
265    pub(crate) fn trait_impls_in_deps_query(
266        db: &dyn HirDatabase,
267        krate: Crate,
268    ) -> Arc<[Arc<Self>]> {
269        let _p = tracing::info_span!("trait_impls_in_deps_query", ?krate).entered();
270        Arc::from_iter(
271            db.transitive_deps(krate).into_iter().map(|krate| db.trait_impls_in_crate(krate)),
272        )
273    }
274
275    fn finish(map: TraitFpMapCollector) -> TraitImpls {
276        TraitImpls {
277            map: map
278                .into_iter()
279                .map(|(k, v)| (k, v.into_iter().map(|(k, v)| (k, v.into_boxed_slice())).collect()))
280                .collect(),
281        }
282    }
283
284    fn collect_def_map(db: &dyn HirDatabase, map: &mut TraitFpMapCollector, def_map: &DefMap) {
285        for (_module_id, module_data) in def_map.modules() {
286            for impl_id in module_data.scope.impls() {
287                // Reservation impls should be ignored during trait resolution, so we never need
288                // them during type analysis. See rust-lang/rust#64631 for details.
289                //
290                // FIXME: Reservation impls should be considered during coherence checks. If we are
291                // (ever) to implement coherence checks, this filtering should be done by the trait
292                // solver.
293                if db.attrs(impl_id.into()).by_key(sym::rustc_reservation_impl).exists() {
294                    continue;
295                }
296                let target_trait = match db.impl_trait(impl_id) {
297                    Some(tr) => tr.skip_binders().hir_trait_id(),
298                    None => continue,
299                };
300                let self_ty = db.impl_self_ty(impl_id);
301                let self_ty_fp = TyFingerprint::for_trait_impl(self_ty.skip_binders());
302                map.entry(target_trait).or_default().entry(self_ty_fp).or_default().push(impl_id);
303            }
304
305            // To better support custom derives, collect impls in all unnamed const items.
306            // const _: () = { ... };
307            for konst in module_data.scope.unnamed_consts() {
308                let body = db.body(konst.into());
309                for (_, block_def_map) in body.blocks(db) {
310                    Self::collect_def_map(db, map, block_def_map);
311                }
312            }
313        }
314    }

358/// Inherent impls defined in some crate.
359///
360/// Inherent impls can only be defined in the crate that also defines the self type of the impl
361/// (note that some primitives are considered to be defined by both libcore and liballoc).
362///
363/// This makes inherent impl lookup easier than trait impl lookup since we only have to consider a
364/// single crate.
365#[derive(Debug, Eq, PartialEq)]
366pub struct InherentImpls {
367    map: FxHashMap<TyFingerprint, Vec<ImplId>>,
368    invalid_impls: Vec<ImplId>,
369}
370
371impl InherentImpls {
372    pub(crate) fn inherent_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
373        let _p = tracing::info_span!("inherent_impls_in_crate_query", ?krate).entered();
374        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
375
376        let crate_def_map = crate_def_map(db, krate);
377        impls.collect_def_map(db, crate_def_map);
378        impls.shrink_to_fit();
379
380        Arc::new(impls)
381    }
382
383    pub(crate) fn inherent_impls_in_block_query(
384        db: &dyn HirDatabase,
385        block: BlockId,
386    ) -> Option<Arc<Self>> {
387        let _p = tracing::info_span!("inherent_impls_in_block_query").entered();
388        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
389
390        let block_def_map = block_def_map(db, block);
391        impls.collect_def_map(db, block_def_map);
392        impls.shrink_to_fit();
393
394        if impls.map.is_empty() && impls.invalid_impls.is_empty() {
395            None
396        } else {
397            Some(Arc::new(impls))
398        }
399    }
400
401    fn shrink_to_fit(&mut self) {
402        self.map.values_mut().for_each(Vec::shrink_to_fit);
403        self.map.shrink_to_fit();
404    }
405
406    fn collect_def_map(&mut self, db: &dyn HirDatabase, def_map: &DefMap) {
407        for (_module_id, module_data) in def_map.modules() {
408            for impl_id in module_data.scope.impls() {
409                let data = db.impl_signature(impl_id);
410                if data.target_trait.is_some() {
411                    continue;
412                }
413
414                let self_ty = db.impl_self_ty(impl_id);
415                let self_ty = self_ty.skip_binders();
416
417                match is_inherent_impl_coherent(db, def_map, impl_id, self_ty) {
418                    true => {
419                        // `fp` should only be `None` in error cases (either erroneous code or incomplete name resolution)
420                        if let Some(fp) = TyFingerprint::for_inherent_impl(self_ty) {
421                            self.map.entry(fp).or_default().push(impl_id);
422                        }
423                    }
424                    false => self.invalid_impls.push(impl_id),
425                }
426            }
427
428            // To better support custom derives, collect impls in all unnamed const items.
429            // const _: () = { ... };
430            for konst in module_data.scope.unnamed_consts() {
431                let body = db.body(konst.into());
432                for (_, block_def_map) in body.blocks(db) {
433                    self.collect_def_map(db, block_def_map);
434                }
435            }
436        }
437    }
438
439    pub fn for_self_ty(&self, self_ty: &Ty) -> &[ImplId] {
440        match TyFingerprint::for_inherent_impl(self_ty) {
441            Some(fp) => self.map.get(&fp).map(|vec| vec.as_ref()).unwrap_or(&[]),
442            None => &[],
443        }
444    }
445
446    pub fn all_impls(&self) -> impl Iterator<Item = ImplId> + '_ {
447        self.map.values().flat_map(|v| v.iter().copied())
448    }
449
450    pub fn invalid_impls(&self) -> &[ImplId] {
451        &self.invalid_impls
452    }
453}

455pub(crate) fn incoherent_inherent_impl_crates(
456    db: &dyn HirDatabase,
457    krate: Crate,
458    fp: TyFingerprint,
459) -> SmallVec<[Crate; 2]> {
460    let _p = tracing::info_span!("incoherent_inherent_impl_crates").entered();
461    let mut res = SmallVec::new();
462
463    // should pass crate for finger print and do reverse deps
464
465    for krate in db.transitive_deps(krate) {
466        let impls = db.inherent_impls_in_crate(krate);
467        if impls.map.get(&fp).is_some_and(|v| !v.is_empty()) {
468            res.push(krate);
469        }
470    }
471
472    res
473}

475pub fn def_crates(db: &dyn HirDatabase, ty: &Ty, cur_crate: Crate) -> Option<SmallVec<[Crate; 2]>> {
476    match ty.kind(Interner) {
477        &TyKind::Adt(AdtId(def_id), _) => {
478            let rustc_has_incoherent_inherent_impls = match def_id {
479                hir_def::AdtId::StructId(id) => db
480                    .struct_signature(id)
481                    .flags
482                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
483                hir_def::AdtId::UnionId(id) => db
484                    .union_signature(id)
485                    .flags
486                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
487                hir_def::AdtId::EnumId(id) => db
488                    .enum_signature(id)
489                    .flags
490                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
491            };
492            Some(if rustc_has_incoherent_inherent_impls {
493                db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Adt(def_id))
494            } else {
495                smallvec![def_id.module(db).krate()]
496            })
497        }
498        &TyKind::Foreign(id) => {
499            let alias = from_foreign_def_id(id);
500            Some(
501                if db
502                    .type_alias_signature(alias)
503                    .flags
504                    .contains(TypeAliasFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPL)
505                {
506                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::ForeignType(id))
507                } else {
508                    smallvec![alias.module(db).krate()]
509                },
510            )
511        }
512        TyKind::Dyn(_) => {
513            let trait_id = ty.dyn_trait()?;
514            Some(
515                if db
516                    .trait_signature(trait_id)
517                    .flags
518                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
519                {
520                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Dyn(trait_id))
521                } else {
522                    smallvec![trait_id.module(db).krate()]
523                },
524            )
525        }
526        // for primitives, there may be impls in various places (core and alloc
527        // mostly). We just check the whole crate graph for crates with impls
528        // (cached behind a query).
529        TyKind::Scalar(_)
530        | TyKind::Str
531        | TyKind::Slice(_)
532        | TyKind::Array(..)
533        | TyKind::Raw(..) => Some(db.incoherent_inherent_impl_crates(
534            cur_crate,
535            TyFingerprint::for_inherent_impl(ty).expect("fingerprint for primitive"),
536        )),
537        _ => None,
538    }
539}

542pub(crate) fn lookup_method<'db>(
543    db: &'db dyn HirDatabase,
544    ty: &next_solver::Canonical<'db, crate::next_solver::Ty<'db>>,

547    visible_from_module: VisibleFromModule,
548    name: &Name,

571/// Whether we're looking up a dotted method call (like `v.len()`) or a path
572/// (like `Vec::new`).
573#[derive(Copy, Clone, Debug, PartialEq, Eq)]
574pub enum LookupMode {
575    /// Looking up a method call like `v.len()`: We only consider candidates
576    /// that have a `self` parameter, and do autoderef.
577    MethodCall,
578    /// Looking up a path like `Vec::new` or `Vec::default`: We consider all
579    /// candidates including associated constants, but don't do autoderef.
580    Path,
581}
582
583#[derive(Clone, Copy)]
584pub enum VisibleFromModule {
585    /// Filter for results that are visible from the given module
586    Filter(ModuleId),
587    /// Include impls from the given block.
588    IncludeBlock(BlockId),
589    /// Do nothing special in regards visibility
590    None,
591}
592
593impl From<Option<ModuleId>> for VisibleFromModule {
594    fn from(module: Option<ModuleId>) -> Self {
595        match module {
596            Some(module) => Self::Filter(module),
597            None => Self::None,
598        }
599    }
600}

602impl From<Option<BlockId>> for VisibleFromModule {
603    fn from(block: Option<BlockId>) -> Self {
604        match block {
605            Some(block) => Self::IncludeBlock(block),
606            None => Self::None,
607        }
608    }

611#[derive(Debug, Clone)]
612pub enum AutorefOrPtrAdjustment {
613    Autoref(Mutability),
614    ToConstPtr,
615}
616
617#[derive(Debug, Clone, Default)]
618pub struct ReceiverAdjustments {
619    autoref: Option<AutorefOrPtrAdjustment>,
620    autoderefs: usize,
621    unsize_array: bool,
622}
623
624impl ReceiverAdjustments {
625    pub(crate) fn apply(&self, table: &mut InferenceTable<'_>, ty: Ty) -> (Ty, Vec<Adjustment>) {
626        let mut ty = table.structurally_resolve_type(&ty);
627        let mut adjust = Vec::new();
628        for _ in 0..self.autoderefs {
629            match autoderef::autoderef_step(table, ty.clone(), true, false) {
630                None => {
631                    never!("autoderef not possible for {:?}", ty);
632                    ty = TyKind::Error.intern(Interner);
633                    break;
634                }
635                Some((kind, new_ty)) => {
636                    ty = new_ty.clone();
637                    let mutbl = match self.autoref {
638                        Some(AutorefOrPtrAdjustment::Autoref(m)) => Some(m),
639                        Some(AutorefOrPtrAdjustment::ToConstPtr) => Some(Mutability::Not),
640                        // FIXME should we know the mutability here, when autoref is `None`?
641                        None => None,
642                    };
643                    adjust.push(Adjustment {
644                        kind: Adjust::Deref(match kind {
645                            AutoderefKind::Overloaded => Some(OverloadedDeref(mutbl)),
646                            AutoderefKind::Builtin => None,
647                        }),
648                        target: new_ty,
649                    });
650                }
651            }
652        }
653        if let Some(autoref) = &self.autoref {
654            let lt = table.new_lifetime_var();
655            match autoref {
656                AutorefOrPtrAdjustment::Autoref(m) => {
657                    let a = Adjustment::borrow(*m, ty, lt);
658                    ty = a.target.clone();
659                    adjust.push(a);
660                }
661                AutorefOrPtrAdjustment::ToConstPtr => {
662                    if let TyKind::Raw(Mutability::Mut, pointee) = ty.kind(Interner) {
663                        let a = Adjustment {
664                            kind: Adjust::Pointer(PointerCast::MutToConstPointer),
665                            target: TyKind::Raw(Mutability::Not, pointee.clone()).intern(Interner),
666                        };
667                        ty = a.target.clone();
668                        adjust.push(a);
669                    } else {
670                        never!("`ToConstPtr` target is not a raw mutable pointer");
671                    }
672                }
673            };
674        }
675        if self.unsize_array {
676            ty = 'it: {
677                if let TyKind::Ref(m, l, inner) = ty.kind(Interner)
678                    && let TyKind::Array(inner, _) = inner.kind(Interner)
679                {
680                    break 'it TyKind::Ref(
681                        *m,
682                        l.clone(),
683                        TyKind::Slice(inner.clone()).intern(Interner),
684                    )
685                    .intern(Interner);
686                }
687                // FIXME: report diagnostic if array unsizing happens without indirection.
688                ty
689            };
690            adjust.push(Adjustment {
691                kind: Adjust::Pointer(PointerCast::Unsize),
692                target: ty.clone(),
693            });
694        }
695        (ty, adjust)
696    }
697
698    fn with_autoref(&self, a: AutorefOrPtrAdjustment) -> ReceiverAdjustments {
699        Self { autoref: Some(a), ..*self }
700    }
701}

763/// Checks if the self parameter of `Trait` method is the `dyn Trait` and we should
764/// call the method using the vtable.
765pub fn is_dyn_method(
766    db: &dyn HirDatabase,
767    _env: Arc<TraitEnvironment>,
768    func: FunctionId,
769    fn_subst: Substitution,
770) -> Option<usize> {
771    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
772        return None;
773    };
774    let trait_params = db.generic_params(trait_id.into()).len();
775    let fn_params = fn_subst.len(Interner) - trait_params;
776    let trait_ref = TraitRef {
777        trait_id: to_chalk_trait_id(trait_id),
778        substitution: Substitution::from_iter(Interner, fn_subst.iter(Interner).take(trait_params)),
779    };
780    let self_ty = trait_ref.self_type_parameter(Interner);
781    if let TyKind::Dyn(d) = self_ty.kind(Interner) {
782        let is_my_trait_in_bounds = d
783            .bounds
784            .skip_binders()
785            .as_slice(Interner)
786            .iter()
787            .map(|it| it.skip_binders())
788            .flat_map(|it| match it {
789                WhereClause::Implemented(tr) => {
790                    all_super_traits(db, from_chalk_trait_id(tr.trait_id))
791                }
792                _ => smallvec![],
793            })
794            // rustc doesn't accept `impl Foo<2> for dyn Foo<5>`, so if the trait id is equal, no matter
795            // what the generics are, we are sure that the method is come from the vtable.
796            .any(|x| x == trait_id);
797        if is_my_trait_in_bounds {
798            return Some(fn_params);
799        }
800    }
801    None
802}

804/// Looks up the impl method that actually runs for the trait method `func`.
805///
806/// Returns `func` if it's not a method defined in a trait or the lookup failed.
807pub(crate) fn lookup_impl_method_query(
808    db: &dyn HirDatabase,
809    env: Arc<TraitEnvironment>,
810    func: FunctionId,
811    fn_subst: Substitution,
812) -> (FunctionId, Substitution) {
813    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
814        return (func, fn_subst);
815    };
816    let trait_params = db.generic_params(trait_id.into()).len();
817    let trait_ref = TraitRef {
818        trait_id: to_chalk_trait_id(trait_id),
819        substitution: Substitution::from_iter(Interner, fn_subst.iter(Interner).take(trait_params)),
820    };
821
822    let name = &db.function_signature(func).name;
823    let Some((impl_fn, impl_subst)) =
824        lookup_impl_assoc_item_for_trait_ref(trait_ref, db, env, name).and_then(|assoc| {
825            if let (AssocItemId::FunctionId(id), subst) = assoc { Some((id, subst)) } else { None }
826        })
827    else {
828        return (func, fn_subst);
829    };
830
831    (
832        impl_fn,
833        Substitution::from_iter(
834            Interner,
835            impl_subst.iter(Interner).chain(fn_subst.iter(Interner).skip(trait_params)),
836        ),
837    )
838}

840fn lookup_impl_assoc_item_for_trait_ref(
841    trait_ref: TraitRef,
842    db: &dyn HirDatabase,
843    env: Arc<TraitEnvironment>,
844    name: &Name,
845) -> Option<(AssocItemId, Substitution)> {
846    let hir_trait_id = trait_ref.hir_trait_id();
847    let self_ty = trait_ref.self_type_parameter(Interner);
848    let self_ty_fp = TyFingerprint::for_trait_impl(&self_ty)?;
849    let impls = db.trait_impls_in_deps(env.krate);
850
851    let trait_module = hir_trait_id.module(db);
852    let type_module = match self_ty_fp {
853        TyFingerprint::Adt(adt_id) => Some(adt_id.module(db)),
854        TyFingerprint::ForeignType(type_id) => Some(from_foreign_def_id(type_id).module(db)),
855        TyFingerprint::Dyn(trait_id) => Some(trait_id.module(db)),
856        _ => None,
857    };
858
859    let def_blocks: ArrayVec<_, 2> =
860        [trait_module.containing_block(), type_module.and_then(|it| it.containing_block())]
861            .into_iter()
862            .flatten()
863            .filter_map(|block_id| db.trait_impls_in_block(block_id))
864            .collect();
865
866    let impls = impls
867        .iter()
868        .chain(&def_blocks)
869        .flat_map(|impls| impls.for_trait_and_self_ty(hir_trait_id, self_ty_fp));
870
871    let table = InferenceTable::new(db, env);
872
873    let (impl_data, impl_subst) = find_matching_impl(impls, table, trait_ref)?;
874    let item = impl_data.items.iter().find_map(|(n, it)| match *it {
875        AssocItemId::FunctionId(f) => (n == name).then_some(AssocItemId::FunctionId(f)),
876        AssocItemId::ConstId(c) => (n == name).then_some(AssocItemId::ConstId(c)),
877        AssocItemId::TypeAliasId(_) => None,
878    })?;
879    Some((item, impl_subst))
880}

882fn find_matching_impl(
883    mut impls: impl Iterator<Item = ImplId>,
884    mut table: InferenceTable<'_>,
885    actual_trait_ref: TraitRef,
886) -> Option<(&ImplItems, Substitution)> {
887    let db = table.db;
888    impls.find_map(|impl_| {
889        table.run_in_snapshot(|table| {
890            let impl_substs =
891                TyBuilder::subst_for_def(db, impl_, None).fill_with_inference_vars(table).build();
892            let trait_ref = db
893                .impl_trait(impl_)
894                .expect("non-trait method in find_matching_impl")
895                .substitute(Interner, &impl_substs);
896
897            if !table.unify(&trait_ref, &actual_trait_ref) {
898                return None;
899            }
900
901            let wcs = crate::chalk_db::convert_where_clauses(db, impl_.into(), &impl_substs)
902                .into_iter()
903                .map(|b| -> Goal { b.cast(Interner) });
904            for goal in wcs {
905                if table.try_obligation(goal.clone()).no_solution() {
906                    return None;
907                }
908                table.register_obligation(goal.to_nextsolver(table.interner));
909            }
910            Some((
911                impl_.impl_items(db),
912                table.resolve_completely::<_, crate::next_solver::GenericArgs<'_>>(impl_substs),
913            ))
914        })
915    })
916}

918fn is_inherent_impl_coherent(
919    db: &dyn HirDatabase,
920    def_map: &DefMap,
921    impl_id: ImplId,
922    self_ty: &Ty,
923) -> bool {
924    let self_ty = self_ty.kind(Interner);
925    let impl_allowed = match self_ty {
926        TyKind::Tuple(_, _)
927        | TyKind::FnDef(_, _)
928        | TyKind::Array(_, _)
929        | TyKind::Never
930        | TyKind::Raw(_, _)
931        | TyKind::Ref(_, _, _)
932        | TyKind::Slice(_)
933        | TyKind::Str
934        | TyKind::Scalar(_) => def_map.is_rustc_coherence_is_core(),
935
936        &TyKind::Adt(AdtId(adt), _) => adt.module(db).krate() == def_map.krate(),
937        TyKind::Dyn(it) => it.principal_id().is_some_and(|trait_id| {
938            from_chalk_trait_id(trait_id).module(db).krate() == def_map.krate()
939        }),
940
941        _ => true,
942    };
943    impl_allowed || {
944        let rustc_has_incoherent_inherent_impls = match self_ty {
945            TyKind::Tuple(_, _)
946            | TyKind::FnDef(_, _)
947            | TyKind::Array(_, _)
948            | TyKind::Never
949            | TyKind::Raw(_, _)
950            | TyKind::Ref(_, _, _)
951            | TyKind::Slice(_)
952            | TyKind::Str
953            | TyKind::Scalar(_) => true,
954
955            &TyKind::Adt(AdtId(adt), _) => match adt {
956                hir_def::AdtId::StructId(id) => db
957                    .struct_signature(id)
958                    .flags
959                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
960                hir_def::AdtId::UnionId(id) => db
961                    .union_signature(id)
962                    .flags
963                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
964                hir_def::AdtId::EnumId(it) => db
965                    .enum_signature(it)
966                    .flags
967                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
968            },
969            TyKind::Dyn(it) => it.principal_id().is_some_and(|trait_id| {
970                db.trait_signature(from_chalk_trait_id(trait_id))
971                    .flags
972                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
973            }),
974
975            _ => false,
976        };
977        let items = impl_id.impl_items(db);
978        rustc_has_incoherent_inherent_impls
979            && !items.items.is_empty()
980            && items.items.iter().all(|&(_, assoc)| match assoc {
981                AssocItemId::FunctionId(it) => {
982                    db.function_signature(it).flags.contains(FnFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
983                }
984                AssocItemId::ConstId(it) => {
985                    db.const_signature(it).flags.contains(ConstFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
986                }
987                AssocItemId::TypeAliasId(it) => db
988                    .type_alias_signature(it)
989                    .flags
990                    .contains(TypeAliasFlags::RUSTC_ALLOW_INCOHERENT_IMPL),
991            })
992    }
993}

995/// Checks whether the impl satisfies the orphan rules.
996///
997/// Given `impl<P1..=Pn> Trait<T1..=Tn> for T0`, an `impl`` is valid only if at least one of the following is true:
998/// - Trait is a local trait
999/// - All of
1000///   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
1001///   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
1002pub fn check_orphan_rules(db: &dyn HirDatabase, impl_: ImplId) -> bool {
1003    let substs = TyBuilder::placeholder_subst(db, impl_);
1004    let Some(impl_trait) = db.impl_trait(impl_) else {
1005        // not a trait impl
1006        return true;
1007    };
1008
1009    let local_crate = impl_.lookup(db).container.krate();
1010    let is_local = |tgt_crate| tgt_crate == local_crate;
1011
1012    let trait_ref = impl_trait.substitute(Interner, &substs);
1013    let trait_id = from_chalk_trait_id(trait_ref.trait_id);
1014    if is_local(trait_id.module(db).krate()) {
1015        // trait to be implemented is local
1016        return true;
1017    }
1018
1019    let unwrap_fundamental = |mut ty: Ty| {
1020        // Unwrap all layers of fundamental types with a loop.
1021        loop {
1022            match ty.kind(Interner) {
1023                TyKind::Ref(_, _, referenced) => ty = referenced.clone(),
1024                &TyKind::Adt(AdtId(hir_def::AdtId::StructId(s)), ref subs) => {
1025                    let struct_signature = db.struct_signature(s);
1026                    if struct_signature.flags.contains(StructFlags::FUNDAMENTAL) {
1027                        let next = subs.type_parameters(Interner).next();
1028                        match next {
1029                            Some(it) => ty = it,
1030                            None => break ty,
1031                        }
1032                    } else {
1033                        break ty;
1034                    }
1035                }
1036                _ => break ty,
1037            }
1038        }
1039    };
1040    //   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
1041
1042    // FIXME: param coverage
1043    //   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
1044    let is_not_orphan = trait_ref.substitution.type_parameters(Interner).any(|ty| {
1045        match unwrap_fundamental(ty).kind(Interner) {
1046            &TyKind::Adt(AdtId(id), _) => is_local(id.module(db).krate()),
1047            TyKind::Error => true,
1048            TyKind::Dyn(it) => it
1049                .principal_id()
1050                .is_some_and(|trait_id| is_local(from_chalk_trait_id(trait_id).module(db).krate())),
1051            _ => false,
1052        }
1053    });
1054    #[allow(clippy::let_and_return)]
1055    is_not_orphan
1056}

1080pub fn iterate_method_candidates_dyn<'db>(
1081    ty: &next_solver::Canonical<'db, crate::next_solver::Ty<'db>>,
1082    db: &'db dyn HirDatabase,
1083    env: Arc<TraitEnvironment>,
1084    traits_in_scope: &FxHashSet<TraitId>,
1085    visible_from_module: VisibleFromModule,
1086    name: Option<&Name>,
1087    mode: LookupMode,
1088    callback: &mut dyn MethodCandidateCallback,
1089) -> ControlFlow<()> {
1090    let _p = tracing::info_span!(
1091        "iterate_method_candidates_dyn",
1092        ?mode,
1093        ?name,
1094        traits_in_scope_len = traits_in_scope.len()
1095    )
1096    .entered();
1097
1098    match mode {
1099        LookupMode::MethodCall => {
1100            // For method calls, rust first does any number of autoderef, and
1101            // then one autoref (i.e. when the method takes &self or &mut self).
1102            // Note that when we've got a receiver like &S, even if the method
1103            // we find in the end takes &self, we still do the autoderef step
1104            // (just as rustc does an autoderef and then autoref again).
1105
1106            // We have to be careful about the order we're looking at candidates
1107            // in here. Consider the case where we're resolving `it.clone()`
1108            // where `it: &Vec<_>`. This resolves to the clone method with self
1109            // type `Vec<_>`, *not* `&_`. I.e. we need to consider methods where
1110            // the receiver type exactly matches before cases where we have to
1111            // do autoref. But in the autoderef steps, the `&_` self type comes
1112            // up *before* the `Vec<_>` self type.
1113            //
1114            // On the other hand, we don't want to just pick any by-value method
1115            // before any by-autoref method; it's just that we need to consider
1116            // the methods by autoderef order of *receiver types*, not *self
1117            // types*.
1118
1119            let mut table = InferenceTable::new(db, env);
1120            let ty = table.instantiate_canonical_ns(*ty);
1121            let deref_chain = autoderef_method_receiver(&mut table, ty);
1122
1123            deref_chain.into_iter().try_for_each(|(receiver_ty, adj)| {
1124                iterate_method_candidates_with_autoref(
1125                    &mut table,
1126                    receiver_ty,
1127                    adj,
1128                    traits_in_scope,
1129                    visible_from_module,
1130                    name,
1131                    callback,
1132                )
1133            })
1134        }
1135        LookupMode::Path => {
1136            // No autoderef for path lookups
1137            iterate_method_candidates_for_self_ty(
1138                ty,
1139                db,
1140                env,
1141                traits_in_scope,
1142                visible_from_module,
1143                name,
1144                callback,
1145            )
1146        }
1147    }
1148}

1150#[tracing::instrument(skip_all, fields(name = ?name))]
1151fn iterate_method_candidates_with_autoref<'db>(
1152    table: &mut InferenceTable<'db>,
1153    receiver_ty: next_solver::Canonical<'db, crate::next_solver::Ty<'db>>,
1154    first_adjustment: ReceiverAdjustments,
1155    traits_in_scope: &FxHashSet<TraitId>,
1156    visible_from_module: VisibleFromModule,
1157    name: Option<&Name>,
1158    callback: &mut dyn MethodCandidateCallback,
1159) -> ControlFlow<()> {
1160    let interner = table.interner;
1161
1162    let mut iterate_method_candidates_by_receiver = move |receiver_ty, first_adjustment| {
1163        iterate_method_candidates_by_receiver(
1164            table,
1165            receiver_ty,
1166            first_adjustment,
1167            traits_in_scope,
1168            visible_from_module,
1169            name,
1170            callback,
1171        )
1172    };
1173
1174    let mut maybe_reborrowed = first_adjustment.clone();
1175    if let rustc_type_ir::TyKind::Ref(_, _, m) = receiver_ty.value.kind() {
1176        let m = match m {
1177            rustc_ast_ir::Mutability::Mut => chalk_ir::Mutability::Mut,
1178            rustc_ast_ir::Mutability::Not => chalk_ir::Mutability::Not,
1179        };
1180        // Prefer reborrow of references to move
1181        maybe_reborrowed.autoref = Some(AutorefOrPtrAdjustment::Autoref(m));
1182        maybe_reborrowed.autoderefs += 1;
1183    }
1184
1185    iterate_method_candidates_by_receiver(receiver_ty, maybe_reborrowed)?;
1186
1187    let refed = next_solver::Canonical {
1188        max_universe: receiver_ty.max_universe,
1189        variables: receiver_ty.variables,
1190        value: next_solver::Ty::new_ref(
1191            interner,
1192            next_solver::Region::error(interner),
1193            receiver_ty.value,
1194            rustc_ast_ir::Mutability::Not,
1195        ),
1196    };
1197
1198    iterate_method_candidates_by_receiver(
1199        refed,
1200        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Not)),
1201    )?;
1202
1203    let ref_muted = next_solver::Canonical {
1204        max_universe: receiver_ty.max_universe,
1205        variables: receiver_ty.variables,
1206        value: next_solver::Ty::new_ref(
1207            interner,
1208            next_solver::Region::error(interner),
1209            receiver_ty.value,
1210            rustc_ast_ir::Mutability::Mut,
1211        ),
1212    };
1213
1214    iterate_method_candidates_by_receiver(
1215        ref_muted,
1216        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Mut)),
1217    )?;
1218
1219    if let rustc_type_ir::TyKind::RawPtr(ty, rustc_ast_ir::Mutability::Mut) =
1220        receiver_ty.value.kind()
1221    {
1222        let const_ptr_ty = rustc_type_ir::Canonical {
1223            max_universe: rustc_type_ir::UniverseIndex::ZERO,
1224            value: next_solver::Ty::new_ptr(interner, ty, rustc_ast_ir::Mutability::Not),
1225            variables: receiver_ty.variables,
1226        };
1227        iterate_method_candidates_by_receiver(
1228            const_ptr_ty,
1229            first_adjustment.with_autoref(AutorefOrPtrAdjustment::ToConstPtr),
1230        )?;
1231    }
1232
1233    ControlFlow::Continue(())
1234}

1275#[tracing::instrument(skip_all, fields(name = ?name))]
1276fn iterate_method_candidates_by_receiver<'db>(
1277    table: &mut InferenceTable<'db>,
1278    receiver_ty: next_solver::Canonical<'db, crate::next_solver::Ty<'db>>,
1279    receiver_adjustments: ReceiverAdjustments,
1280    traits_in_scope: &FxHashSet<TraitId>,
1281    visible_from_module: VisibleFromModule,
1282    name: Option<&Name>,
1283    callback: &mut dyn MethodCandidateCallback,
1284) -> ControlFlow<()> {
1285    let receiver_ty = table.instantiate_canonical_ns(receiver_ty);
1286    let receiver_ty: crate::Ty = receiver_ty.to_chalk(table.interner);
1287    // We're looking for methods with *receiver* type receiver_ty. These could
1288    // be found in any of the derefs of receiver_ty, so we have to go through
1289    // that, including raw derefs.
1290    table.run_in_snapshot(|table| {
1291        let mut autoderef =
1292            autoderef::Autoderef::new_no_tracking(table, receiver_ty.clone(), true, true);
1293        while let Some((self_ty, _)) = autoderef.next() {
1294            iterate_inherent_methods(
1295                &self_ty,
1296                autoderef.table,
1297                name,
1298                Some(&receiver_ty),
1299                Some(receiver_adjustments.clone()),
1300                visible_from_module,
1301                LookupMode::MethodCall,
1302                &mut |adjustments, item, is_visible| {
1303                    callback.on_inherent_method(adjustments, item, is_visible)
1304                },
1305            )?
1306        }
1307        ControlFlow::Continue(())
1308    })?;
1309    table.run_in_snapshot(|table| {
1310        let mut autoderef =
1311            autoderef::Autoderef::new_no_tracking(table, receiver_ty.clone(), true, true);
1312        while let Some((self_ty, _)) = autoderef.next() {
1313            if matches!(self_ty.kind(Interner), TyKind::InferenceVar(_, TyVariableKind::General)) {
1314                // don't try to resolve methods on unknown types
1315                return ControlFlow::Continue(());
1316            }
1317
1318            iterate_trait_method_candidates(
1319                &self_ty,
1320                autoderef.table,
1321                traits_in_scope,
1322                name,
1323                Some(&receiver_ty),
1324                Some(receiver_adjustments.clone()),
1325                LookupMode::MethodCall,
1326                &mut |adjustments, item, is_visible| {
1327                    callback.on_trait_method(adjustments, item, is_visible)
1328                },
1329            )?
1330        }
1331        ControlFlow::Continue(())
1332    })
1333}

1373#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1374fn iterate_trait_method_candidates(
1375    self_ty: &Ty,
1376    table: &mut InferenceTable<'_>,
1377    traits_in_scope: &FxHashSet<TraitId>,
1378    name: Option<&Name>,
1379    receiver_ty: Option<&Ty>,
1380    receiver_adjustments: Option<ReceiverAdjustments>,
1381    mode: LookupMode,
1382    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1383) -> ControlFlow<()> {
1384    let db = table.db;
1385
1386    let canonical_self_ty = table.canonicalize(self_ty.clone().to_nextsolver(table.interner));
1387    let TraitEnvironment { krate, .. } = *table.trait_env;
1388
1389    'traits: for &t in traits_in_scope {
1390        let data = db.trait_signature(t);
1391
1392        // Traits annotated with `#[rustc_skip_during_method_dispatch]` are skipped during
1393        // method resolution, if the receiver is an array, and we're compiling for editions before
1394        // 2021.
1395        // This is to make `[a].into_iter()` not break code with the new `IntoIterator` impl for
1396        // arrays.
1397        if data.flags.contains(TraitFlags::SKIP_ARRAY_DURING_METHOD_DISPATCH)
1398            && matches!(self_ty.kind(Interner), TyKind::Array(..))
1399        {
1400            // FIXME: this should really be using the edition of the method name's span, in case it
1401            // comes from a macro
1402            if !krate.data(db).edition.at_least_2021() {
1403                continue;
1404            }
1405        }
1406        if data.flags.contains(TraitFlags::SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH)
1407            && matches!(
1408                self_ty.kind(Interner), TyKind::Adt(AdtId(def), subst)
1409                if is_box(table.db, *def)
1410                    && matches!(subst.at(Interner, 0).assert_ty_ref(Interner).kind(Interner), TyKind::Slice(..))
1411            )
1412        {
1413            // FIXME: this should really be using the edition of the method name's span, in case it
1414            // comes from a macro
1415            if !krate.data(db).edition.at_least_2024() {
1416                continue;
1417            }
1418        }
1419
1420        // we'll be lazy about checking whether the type implements the
1421        // trait, but if we find out it doesn't, we'll skip the rest of the
1422        // iteration
1423        let mut known_implemented = false;
1424        for &(_, item) in t.trait_items(db).items.iter() {
1425            // Don't pass a `visible_from_module` down to `is_valid_candidate`,
1426            // since only inherent methods should be included into visibility checking.
1427            let visible = match is_valid_trait_method_candidate(
1428                table,
1429                t,
1430                name,
1431                receiver_ty,
1432                item,
1433                self_ty,
1434                mode,
1435            ) {
1436                IsValidCandidate::Yes => true,
1437                IsValidCandidate::NotVisible => false,
1438                IsValidCandidate::No => continue,
1439            };
1440            if !known_implemented {
1441                let goal = generic_implements_goal_ns(table, t, canonical_self_ty);
1442                if next_trait_solve_canonical_in_ctxt(&table.infer_ctxt, goal).no_solution() {
1443                    continue 'traits;
1444                }
1445            }
1446            known_implemented = true;
1447            callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1448        }
1449    }
1450    ControlFlow::Continue(())
1451}

1453#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1454fn iterate_inherent_methods(
1455    self_ty: &Ty,
1456    table: &mut InferenceTable<'_>,
1457    name: Option<&Name>,
1458    receiver_ty: Option<&Ty>,
1459    receiver_adjustments: Option<ReceiverAdjustments>,
1460    visible_from_module: VisibleFromModule,
1461    mode: LookupMode,
1462    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1463) -> ControlFlow<()> {
1464    let db = table.db;
1465    let env = table.trait_env.clone();
1466
1467    // For trait object types and placeholder types with trait bounds, the methods of the trait and
1468    // its super traits are considered inherent methods. This matters because these methods have
1469    // higher priority than the other traits' methods, which would be considered in
1470    // `iterate_trait_method_candidates()` only after this function.
1471    match self_ty.kind(Interner) {
1472        TyKind::Placeholder(_) => {
1473            let env = table.trait_env.clone();
1474            let traits = env
1475                .traits_in_scope_from_clauses(self_ty.clone())
1476                .flat_map(|t| all_super_traits(db, t));
1477            iterate_inherent_trait_methods(
1478                self_ty,
1479                table,
1480                name,
1481                receiver_ty,
1482                receiver_adjustments.clone(),
1483                callback,
1484                traits,
1485                mode,
1486            )?;
1487        }
1488        TyKind::Dyn(_) => {
1489            if let Some(principal_trait) = self_ty.dyn_trait() {
1490                let traits = all_super_traits(db, principal_trait);
1491                iterate_inherent_trait_methods(
1492                    self_ty,
1493                    table,
1494                    name,
1495                    receiver_ty,
1496                    receiver_adjustments.clone(),
1497                    callback,
1498                    traits.into_iter(),
1499                    mode,
1500                )?;
1501            }
1502        }
1503        _ => {}
1504    }
1505
1506    let def_crates = match def_crates(db, self_ty, env.krate) {
1507        Some(k) => k,
1508        None => return ControlFlow::Continue(()),
1509    };
1510
1511    let (module, mut block) = match visible_from_module {
1512        VisibleFromModule::Filter(module) => (Some(module), module.containing_block()),
1513        VisibleFromModule::IncludeBlock(block) => (None, Some(block)),
1514        VisibleFromModule::None => (None, None),
1515    };
1516
1517    while let Some(block_id) = block {
1518        if let Some(impls) = db.inherent_impls_in_block(block_id) {
1519            impls_for_self_ty(
1520                &impls,
1521                self_ty,
1522                table,
1523                name,
1524                receiver_ty,
1525                receiver_adjustments.clone(),
1526                module,
1527                callback,
1528            )?;
1529        }
1530
1531        block = block_def_map(db, block_id).parent().and_then(|module| module.containing_block());
1532    }
1533
1534    for krate in def_crates {
1535        let impls = db.inherent_impls_in_crate(krate);
1536        impls_for_self_ty(
1537            &impls,
1538            self_ty,
1539            table,
1540            name,
1541            receiver_ty,
1542            receiver_adjustments.clone(),
1543            module,
1544            callback,
1545        )?;
1546    }
1547    return ControlFlow::Continue(());
1548
1549    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1550    fn iterate_inherent_trait_methods(
1551        self_ty: &Ty,
1552        table: &mut InferenceTable<'_>,
1553        name: Option<&Name>,
1554        receiver_ty: Option<&Ty>,
1555        receiver_adjustments: Option<ReceiverAdjustments>,
1556        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1557        traits: impl Iterator<Item = TraitId>,
1558        mode: LookupMode,
1559    ) -> ControlFlow<()> {
1560        let db = table.db;
1561        for t in traits {
1562            let data = t.trait_items(db);
1563            for &(_, item) in data.items.iter() {
1564                // We don't pass `visible_from_module` as all trait items should be visible.
1565                let visible = match is_valid_trait_method_candidate(
1566                    table,
1567                    t,
1568                    name,
1569                    receiver_ty,
1570                    item,
1571                    self_ty,
1572                    mode,
1573                ) {
1574                    IsValidCandidate::Yes => true,
1575                    IsValidCandidate::NotVisible => false,
1576                    IsValidCandidate::No => continue,
1577                };
1578                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1579            }
1580        }
1581        ControlFlow::Continue(())
1582    }
1583
1584    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1585    fn impls_for_self_ty(
1586        impls: &InherentImpls,
1587        self_ty: &Ty,
1588        table: &mut InferenceTable<'_>,
1589        name: Option<&Name>,
1590        receiver_ty: Option<&Ty>,
1591        receiver_adjustments: Option<ReceiverAdjustments>,
1592        visible_from_module: Option<ModuleId>,
1593        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1594    ) -> ControlFlow<()> {
1595        for &impl_id in impls.for_self_ty(self_ty) {
1596            for &(ref item_name, item) in impl_id.impl_items(table.db).items.iter() {
1597                let visible = match is_valid_impl_method_candidate(
1598                    table,
1599                    self_ty,
1600                    receiver_ty,
1601                    visible_from_module,
1602                    name,
1603                    impl_id,
1604                    item,
1605                    item_name,
1606                ) {
1607                    IsValidCandidate::Yes => true,
1608                    IsValidCandidate::NotVisible => false,
1609                    IsValidCandidate::No => continue,
1610                };
1611                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1612            }
1613        }
1614        ControlFlow::Continue(())
1615    }
1616}

1652#[tracing::instrument(skip_all, fields(name))]
1653fn is_valid_impl_method_candidate(
1654    table: &mut InferenceTable<'_>,
1655    self_ty: &Ty,
1656    receiver_ty: Option<&Ty>,
1657    visible_from_module: Option<ModuleId>,
1658    name: Option<&Name>,
1659    impl_id: ImplId,
1660    item: AssocItemId,
1661    item_name: &Name,
1662) -> IsValidCandidate {
1663    match item {
1664        AssocItemId::FunctionId(f) => is_valid_impl_fn_candidate(
1665            table,
1666            impl_id,
1667            f,
1668            name,
1669            receiver_ty,
1670            self_ty,
1671            visible_from_module,
1672            item_name,
1673        ),
1674        AssocItemId::ConstId(c) => {
1675            let db = table.db;
1676            check_that!(receiver_ty.is_none());
1677            check_that!(name.is_none_or(|n| n == item_name));
1678
1679            if let Some(from_module) = visible_from_module
1680                && !db.assoc_visibility(c.into()).is_visible_from(db, from_module)
1681            {
1682                cov_mark::hit!(const_candidate_not_visible);
1683                return IsValidCandidate::NotVisible;
1684            }
1685            let self_ty_matches = table.run_in_snapshot(|table| {
1686                let expected_self_ty =
1687                    TyBuilder::impl_self_ty(db, impl_id).fill_with_inference_vars(table).build();
1688                table.unify(&expected_self_ty, self_ty)
1689            });
1690            if !self_ty_matches {
1691                cov_mark::hit!(const_candidate_self_type_mismatch);
1692                return IsValidCandidate::No;
1693            }
1694            IsValidCandidate::Yes
1695        }
1696        _ => IsValidCandidate::No,
1697    }
1698}

1700/// Checks whether a given `AssocItemId` is applicable for `receiver_ty`.
1701#[tracing::instrument(skip_all, fields(name))]
1702fn is_valid_trait_method_candidate(
1703    table: &mut InferenceTable<'_>,
1704    trait_id: TraitId,
1705    name: Option<&Name>,
1706    receiver_ty: Option<&Ty>,
1707    item: AssocItemId,
1708    self_ty: &Ty,
1709    mode: LookupMode,
1710) -> IsValidCandidate {
1711    let db = table.db;
1712    match item {
1713        AssocItemId::FunctionId(fn_id) => {
1714            let data = db.function_signature(fn_id);
1715
1716            check_that!(name.is_none_or(|n| n == &data.name));
1717
1718            table.run_in_snapshot(|table| {
1719                let impl_subst = TyBuilder::subst_for_def(db, trait_id, None)
1720                    .fill_with_inference_vars(table)
1721                    .build();
1722                let expect_self_ty = impl_subst.at(Interner, 0).assert_ty_ref(Interner).clone();
1723
1724                check_that!(table.unify(&expect_self_ty, self_ty));
1725
1726                if let Some(receiver_ty) = receiver_ty {
1727                    check_that!(data.has_self_param());
1728
1729                    let fn_subst = TyBuilder::subst_for_def(db, fn_id, Some(impl_subst))
1730                        .fill_with_inference_vars(table)
1731                        .build();
1732
1733                    let sig = db.callable_item_signature(fn_id.into());
1734                    let expected_receiver =
1735                        sig.map(|s| s.params()[0].clone()).substitute(Interner, &fn_subst);
1736
1737                    // FIXME: Clean up this mess with some context struct like rustc's `ProbeContext`
1738                    let variance = match mode {
1739                        LookupMode::MethodCall => rustc_type_ir::Variance::Covariant,
1740                        LookupMode::Path => rustc_type_ir::Variance::Invariant,
1741                    };
1742                    let res = table
1743                        .infer_ctxt
1744                        .at(
1745                            &next_solver::infer::traits::ObligationCause::dummy(),
1746                            table.trait_env.env.to_nextsolver(table.interner),
1747                        )
1748                        .relate(
1749                            DefineOpaqueTypes::No,
1750                            expected_receiver.to_nextsolver(table.interner),
1751                            variance,
1752                            receiver_ty.to_nextsolver(table.interner),
1753                        );
1754                    let Ok(infer_ok) = res else {
1755                        return IsValidCandidate::No;
1756                    };
1757
1758                    if !infer_ok.obligations.is_empty() {
1759                        let mut ctxt = FulfillmentCtxt::new(&table.infer_ctxt);
1760                        for pred in infer_ok.into_obligations() {
1761                            ctxt.register_predicate_obligation(&table.infer_ctxt, pred);
1762                        }
1763                        check_that!(ctxt.select_all_or_error(&table.infer_ctxt).is_empty());
1764                    }
1765
1766                    check_that!(table.unify(receiver_ty, &expected_receiver));
1767                }
1768
1769                IsValidCandidate::Yes
1770            })
1771        }
1772        AssocItemId::ConstId(c) => {
1773            check_that!(receiver_ty.is_none());
1774            check_that!(name.is_none_or(|n| db.const_signature(c).name.as_ref() == Some(n)));
1775
1776            IsValidCandidate::Yes
1777        }
1778        _ => IsValidCandidate::No,
1779    }
1780}

1782#[tracing::instrument(skip_all, fields(name))]
1783fn is_valid_impl_fn_candidate(
1784    table: &mut InferenceTable<'_>,
1785    impl_id: ImplId,
1786    fn_id: FunctionId,
1787    name: Option<&Name>,
1788    receiver_ty: Option<&Ty>,
1789    self_ty: &Ty,
1790    visible_from_module: Option<ModuleId>,
1791    item_name: &Name,
1792) -> IsValidCandidate {
1793    check_that!(name.is_none_or(|n| n == item_name));
1794
1795    let db = table.db;
1796    let data = db.function_signature(fn_id);
1797
1798    if let Some(from_module) = visible_from_module
1799        && !db.assoc_visibility(fn_id.into()).is_visible_from(db, from_module)
1800    {
1801        cov_mark::hit!(autoderef_candidate_not_visible);
1802        return IsValidCandidate::NotVisible;
1803    }
1804    table.run_in_snapshot(|table| {
1805        let _p = tracing::info_span!("subst_for_def").entered();
1806        let impl_subst =
1807            TyBuilder::subst_for_def(db, impl_id, None).fill_with_inference_vars(table).build();
1808        let expect_self_ty = db.impl_self_ty(impl_id).substitute(Interner, &impl_subst);
1809
1810        check_that!(table.unify(&expect_self_ty, self_ty));
1811
1812        if let Some(receiver_ty) = receiver_ty {
1813            let _p = tracing::info_span!("check_receiver_ty").entered();
1814            check_that!(data.has_self_param());
1815
1816            let fn_subst = TyBuilder::subst_for_def(db, fn_id, Some(impl_subst.clone()))
1817                .fill_with_inference_vars(table)
1818                .build();
1819
1820            let sig = db.callable_item_signature(fn_id.into());
1821            let expected_receiver =
1822                sig.map(|s| s.params()[0].clone()).substitute(Interner, &fn_subst);
1823
1824            check_that!(table.unify(receiver_ty, &expected_receiver));
1825        }
1826
1827        // We need to consider the bounds on the impl to distinguish functions of the same name
1828        // for a type.
1829        let predicates = db.generic_predicates(impl_id.into());
1830        let goals = predicates.iter().map(|p| {
1831            let (p, b) = p
1832                .clone()
1833                .substitute(Interner, &impl_subst)
1834                // Skipping the inner binders is ok, as we don't handle quantified where
1835                // clauses yet.
1836                .into_value_and_skipped_binders();
1837            stdx::always!(b.len(Interner) == 0);
1838
1839            p.cast::<Goal>(Interner)
1840        });
1841
1842        for goal in goals.clone() {
1843            match table.solve_obligation(goal) {
1844                Ok(_) => {}
1845                Err(_) => {
1846                    return IsValidCandidate::No;
1847                }
1848            }
1849        }
1850
1851        for goal in goals {
1852            if table.try_obligation(goal).no_solution() {
1853                return IsValidCandidate::No;
1854            }
1855        }
1856
1857        IsValidCandidate::Yes
1858    })
1859}

1881/// This creates Substs for a trait with the given Self type and type variables
1882/// for all other parameters, to query Chalk with it.
1883#[tracing::instrument(skip_all)]
1884fn generic_implements_goal(
1885    db: &dyn HirDatabase,
1886    env: &TraitEnvironment,
1887    trait_: TraitId,
1888    self_ty: &Canonical<Ty>,
1889) -> Canonical<InEnvironment<super::DomainGoal>> {
1890    let binders = self_ty.binders.interned();
1891    let trait_ref = TyBuilder::trait_ref(db, trait_)
1892        .push(self_ty.value.clone())
1893        .fill_with_bound_vars(DebruijnIndex::INNERMOST, binders.len())
1894        .build();
1895
1896    let kinds =
1897        binders.iter().cloned().chain(trait_ref.substitution.iter(Interner).skip(1).map(|it| {
1898            let vk = match it.data(Interner) {
1899                GenericArgData::Ty(_) => VariableKind::Ty(chalk_ir::TyVariableKind::General),
1900                GenericArgData::Lifetime(_) => VariableKind::Lifetime,
1901                GenericArgData::Const(c) => VariableKind::Const(c.data(Interner).ty.clone()),
1902            };
1903            WithKind::new(vk, UniverseIndex::ROOT)
1904        }));
1905    let binders = CanonicalVarKinds::from_iter(Interner, kinds);
1906
1907    let obligation = trait_ref.cast(Interner);
1908    let value = InEnvironment::new(&env.env, obligation);
1909    Canonical { binders, value }
1910}

1912/// This creates Substs for a trait with the given Self type and type variables
1913/// for all other parameters, to query the trait solver with it.
1914#[tracing::instrument(skip_all)]
1915fn generic_implements_goal_ns<'db>(
1916    table: &mut InferenceTable<'db>,
1917    trait_: TraitId,
1918    self_ty: next_solver::Canonical<'db, crate::next_solver::Ty<'db>>,
1919) -> next_solver::Canonical<'db, next_solver::Goal<'db, crate::next_solver::Predicate<'db>>> {
1920    let args = table.infer_ctxt.fresh_args_for_item(SolverDefId::TraitId(trait_));
1921    let self_ty = table.instantiate_canonical_ns(self_ty);
1922    let trait_ref =
1923        rustc_type_ir::TraitRef::new_from_args(table.infer_ctxt.interner, trait_.into(), args)
1924            .with_replaced_self_ty(table.infer_ctxt.interner, self_ty);
1925    let goal = next_solver::Goal::new(
1926        table.infer_ctxt.interner,
1927        table.trait_env.env.to_nextsolver(table.infer_ctxt.interner),
1928        trait_ref,
1929    );
1930
1931    table.canonicalize(goal)
1932}

1934fn autoderef_method_receiver<'db>(
1935    table: &mut InferenceTable<'db>,
1936    ty: next_solver::Ty<'db>,
1937) -> Vec<(next_solver::Canonical<'db, crate::next_solver::Ty<'db>>, ReceiverAdjustments)> {
1938    let interner = table.interner;
1939    let mut deref_chain = Vec::new();
1940    let mut autoderef =
1941        autoderef::Autoderef::new_no_tracking(table, ty.to_chalk(interner), false, true);
1942    while let Some((ty, derefs)) = autoderef.next() {
1943        deref_chain.push((
1944            autoderef.table.canonicalize(ty.to_nextsolver(interner)),
1945            ReceiverAdjustments { autoref: None, autoderefs: derefs, unsize_array: false },
1946        ));
1947    }
1948    // As a last step, we can do array unsizing (that's the only unsizing that rustc does for method receivers!)
1949    if let Some((rustc_type_ir::Array(parameters, _), variables, max_universe, adj)) =
1950        deref_chain.last().map(|d| (d.0.value.kind(), d.0.variables, d.0.max_universe, d.1.clone()))
1951    {
1952        let unsized_ty = next_solver::Ty::new_slice(interner, parameters);
1953        deref_chain.push((
1954            next_solver::Canonical { max_universe, value: unsized_ty, variables },
1955            ReceiverAdjustments { unsize_array: true, ..adj.clone() },
1956        ));
1957    }
1958    deref_chain
1959}
