39#[derive(Copy, Clone, Debug)]
40pub struct OpaqueHiddenType<'db> {
41    pub ty: Ty<'db>,
42}
43
44impl<'db> InferCtxt<'db> {
45    /// Insert a hidden type into the opaque type storage, making sure
46    /// it hasn't previously been defined. This does not emit any
47    /// constraints and it's the responsibility of the caller to make
48    /// sure that the item bounds of the opaque are checked.
49    pub fn register_hidden_type_in_storage(
50        &self,
51        opaque_type_key: OpaqueTypeKey<'db>,
52        hidden_ty: OpaqueHiddenType<'db>,
53    ) -> Option<Ty<'db>> {
54        self.inner.borrow_mut().opaque_types().register(opaque_type_key, hidden_ty)
55    }
56}
