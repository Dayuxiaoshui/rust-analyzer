144    /// Fast path helper for testing if a type is `Sized` or `MetaSized`.
145    ///
146    /// Returning true means the type is known to implement the sizedness trait. Returning `false`
147    /// means nothing -- could be sized, might not be.
148    ///
149    /// Note that we could never rely on the fact that a type such as `[_]` is trivially `!Sized`
150    /// because we could be in a type environment with a bound such as `[_]: Copy`. A function with
151    /// such a bound obviously never can be called, but that doesn't mean it shouldn't typecheck.
152    /// This is why this method doesn't return `Option<bool>`.
153    #[tracing::instrument(skip(tcx), level = "debug")]
154    pub fn has_trivial_sizedness(self, tcx: DbInterner<'db>, sizedness: SizedTraitKind) -> bool {
155        match self.kind() {
156            TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
157            | TyKind::Uint(_)
158            | TyKind::Int(_)
159            | TyKind::Bool
160            | TyKind::Float(_)
161            | TyKind::FnDef(..)
162            | TyKind::FnPtr(..)
163            | TyKind::UnsafeBinder(_)
164            | TyKind::RawPtr(..)
165            | TyKind::Char
166            | TyKind::Ref(..)
167            | TyKind::Coroutine(..)
168            | TyKind::CoroutineWitness(..)
169            | TyKind::Array(..)
170            | TyKind::Pat(..)
171            | TyKind::Closure(..)
172            | TyKind::CoroutineClosure(..)
173            | TyKind::Never
174            | TyKind::Error(_) => true,
175
176            TyKind::Str | TyKind::Slice(_) | TyKind::Dynamic(_, _, _) => match sizedness {
177                SizedTraitKind::Sized => false,
178                SizedTraitKind::MetaSized => true,
179            },
180
181            TyKind::Foreign(..) => match sizedness {
182                SizedTraitKind::Sized | SizedTraitKind::MetaSized => false,
183            },
184
185            TyKind::Tuple(tys) => {
186                tys.last().is_none_or(|ty| ty.has_trivial_sizedness(tcx, sizedness))
187            }
188
189            TyKind::Adt(def, args) => def
190                .sizedness_constraint(tcx, sizedness)
191                .is_none_or(|ty| ty.instantiate(tcx, args).has_trivial_sizedness(tcx, sizedness)),
192
193            TyKind::Alias(..) | TyKind::Param(_) | TyKind::Placeholder(..) | TyKind::Bound(..) => {
194                false
195            }
196
197            TyKind::Infer(InferTy::TyVar(_)) => false,
198
199            TyKind::Infer(
200                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
201            ) => {
202                panic!("`has_trivial_sizedness` applied to unexpected type: {self:?}")
203            }
204        }
205    }

207    /// Fast path helper for primitives which are always `Copy` and which
208    /// have a side-effect-free `Clone` impl.
209    ///
210    /// Returning true means the type is known to be pure and `Copy+Clone`.
211    /// Returning `false` means nothing -- could be `Copy`, might not be.
212    ///
213    /// This is mostly useful for optimizations, as these are the types
214    /// on which we can replace cloning with dereferencing.
215    pub fn is_trivially_pure_clone_copy(self) -> bool {
216        match self.kind() {
217            TyKind::Bool | TyKind::Char | TyKind::Never => true,
218
219            // These aren't even `Clone`
220            TyKind::Str | TyKind::Slice(..) | TyKind::Foreign(..) | TyKind::Dynamic(..) => false,
221
222            TyKind::Infer(InferTy::FloatVar(_) | InferTy::IntVar(_))
223            | TyKind::Int(..)
224            | TyKind::Uint(..)
225            | TyKind::Float(..) => true,
226
227            // ZST which can't be named are fine.
228            TyKind::FnDef(..) => true,
229
230            TyKind::Array(element_ty, _len) => element_ty.is_trivially_pure_clone_copy(),
231
232            // A 100-tuple isn't "trivial", so doing this only for reasonable sizes.
233            TyKind::Tuple(field_tys) => {
234                field_tys.len() <= 3 && field_tys.iter().all(Self::is_trivially_pure_clone_copy)
235            }
236
237            TyKind::Pat(ty, _) => ty.is_trivially_pure_clone_copy(),
238
239            // Sometimes traits aren't implemented for every ABI or arity,
240            // because we can't be generic over everything yet.
241            TyKind::FnPtr(..) => false,
242
243            // Definitely absolutely not copy.
244            TyKind::Ref(_, _, Mutability::Mut) => false,
245
246            // The standard library has a blanket Copy impl for shared references and raw pointers,
247            // for all unsized types.
248            TyKind::Ref(_, _, Mutability::Not) | TyKind::RawPtr(..) => true,
249
250            TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => false,
251
252            // Might be, but not "trivial" so just giving the safe answer.
253            TyKind::Adt(..) | TyKind::Closure(..) | TyKind::CoroutineClosure(..) => false,
254
255            TyKind::UnsafeBinder(_) => false,
256
257            // Needs normalization or revealing to determine, so no is the safe answer.
258            TyKind::Alias(..) => false,
259
260            TyKind::Param(..)
261            | TyKind::Placeholder(..)
262            | TyKind::Bound(..)
263            | TyKind::Infer(..)
264            | TyKind::Error(..) => false,
265        }
266    }

268    pub fn is_trivially_wf(self, tcx: DbInterner<'db>) -> bool {
269        match self.kind() {
270            TyKind::Bool
271            | TyKind::Char
272            | TyKind::Int(_)
273            | TyKind::Uint(_)
274            | TyKind::Float(_)
275            | TyKind::Str
276            | TyKind::Never
277            | TyKind::Param(_)
278            | TyKind::Placeholder(_)
279            | TyKind::Bound(..) => true,
280
281            TyKind::Slice(ty) => {
282                ty.is_trivially_wf(tcx) && ty.has_trivial_sizedness(tcx, SizedTraitKind::Sized)
283            }
284            TyKind::RawPtr(ty, _) => ty.is_trivially_wf(tcx),
285
286            TyKind::FnPtr(sig_tys, _) => {
287                sig_tys.skip_binder().inputs_and_output.iter().all(|ty| ty.is_trivially_wf(tcx))
288            }
289            TyKind::Ref(_, ty, _) => ty.is_global() && ty.is_trivially_wf(tcx),
290
291            TyKind::Infer(infer) => match infer {
292                InferTy::TyVar(_) => false,
293                InferTy::IntVar(_) | InferTy::FloatVar(_) => true,
294                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => true,
295            },
296
297            TyKind::Adt(_, _)
298            | TyKind::Tuple(_)
299            | TyKind::Array(..)
300            | TyKind::Foreign(_)
301            | TyKind::Pat(_, _)
302            | TyKind::FnDef(..)
303            | TyKind::UnsafeBinder(..)
304            | TyKind::Dynamic(..)
305            | TyKind::Closure(..)
306            | TyKind::CoroutineClosure(..)
307            | TyKind::Coroutine(..)
308            | TyKind::CoroutineWitness(..)
309            | TyKind::Alias(..)
310            | TyKind::Error(_) => false,
311        }
312    }

344impl<'db> TypeSuperVisitable<DbInterner<'db>> for Ty<'db> {
345    fn super_visit_with<V: rustc_type_ir::TypeVisitor<DbInterner<'db>>>(
346        &self,
347        visitor: &mut V,
348    ) -> V::Result {
349        match (*self).kind() {
350            TyKind::RawPtr(ty, _mutbl) => ty.visit_with(visitor),
351            TyKind::Array(typ, sz) => {
352                try_visit!(typ.visit_with(visitor));
353                sz.visit_with(visitor)
354            }
355            TyKind::Slice(typ) => typ.visit_with(visitor),
356            TyKind::Adt(_, args) => args.visit_with(visitor),
357            TyKind::Dynamic(ref trait_ty, ref reg, _) => {
358                try_visit!(trait_ty.visit_with(visitor));
359                reg.visit_with(visitor)
360            }
361            TyKind::Tuple(ts) => ts.visit_with(visitor),
362            TyKind::FnDef(_, args) => args.visit_with(visitor),
363            TyKind::FnPtr(ref sig_tys, _) => sig_tys.visit_with(visitor),
364            TyKind::UnsafeBinder(f) => f.visit_with(visitor),
365            TyKind::Ref(r, ty, _) => {
366                try_visit!(r.visit_with(visitor));
367                ty.visit_with(visitor)
368            }
369            TyKind::Coroutine(_did, ref args) => args.visit_with(visitor),
370            TyKind::CoroutineWitness(_did, ref args) => args.visit_with(visitor),
371            TyKind::Closure(_did, ref args) => args.visit_with(visitor),
372            TyKind::CoroutineClosure(_did, ref args) => args.visit_with(visitor),
373            TyKind::Alias(_, ref data) => data.visit_with(visitor),
374
375            TyKind::Pat(ty, pat) => {
376                try_visit!(ty.visit_with(visitor));
377                pat.visit_with(visitor)
378            }
379
380            TyKind::Error(guar) => guar.visit_with(visitor),
381
382            TyKind::Bool
383            | TyKind::Char
384            | TyKind::Str
385            | TyKind::Int(_)
386            | TyKind::Uint(_)
387            | TyKind::Float(_)
388            | TyKind::Infer(_)
389            | TyKind::Bound(..)
390            | TyKind::Placeholder(..)
391            | TyKind::Param(..)
392            | TyKind::Never
393            | TyKind::Foreign(..) => V::Result::output(),
394        }
395    }
396}

410impl<'db> TypeSuperFoldable<DbInterner<'db>> for Ty<'db> {
411    fn try_super_fold_with<F: rustc_type_ir::FallibleTypeFolder<DbInterner<'db>>>(
412        self,
413        folder: &mut F,
414    ) -> Result<Self, F::Error> {
415        let kind = match self.kind() {
416            TyKind::RawPtr(ty, mutbl) => TyKind::RawPtr(ty.try_fold_with(folder)?, mutbl),
417            TyKind::Array(typ, sz) => {
418                TyKind::Array(typ.try_fold_with(folder)?, sz.try_fold_with(folder)?)
419            }
420            TyKind::Slice(typ) => TyKind::Slice(typ.try_fold_with(folder)?),
421            TyKind::Adt(tid, args) => TyKind::Adt(tid, args.try_fold_with(folder)?),
422            TyKind::Dynamic(trait_ty, region, representation) => TyKind::Dynamic(
423                trait_ty.try_fold_with(folder)?,
424                region.try_fold_with(folder)?,
425                representation,
426            ),
427            TyKind::Tuple(ts) => TyKind::Tuple(ts.try_fold_with(folder)?),
428            TyKind::FnDef(def_id, args) => TyKind::FnDef(def_id, args.try_fold_with(folder)?),
429            TyKind::FnPtr(sig_tys, hdr) => TyKind::FnPtr(sig_tys.try_fold_with(folder)?, hdr),
430            TyKind::UnsafeBinder(f) => TyKind::UnsafeBinder(f.try_fold_with(folder)?),
431            TyKind::Ref(r, ty, mutbl) => {
432                TyKind::Ref(r.try_fold_with(folder)?, ty.try_fold_with(folder)?, mutbl)
433            }
434            TyKind::Coroutine(did, args) => TyKind::Coroutine(did, args.try_fold_with(folder)?),
435            TyKind::CoroutineWitness(did, args) => {
436                TyKind::CoroutineWitness(did, args.try_fold_with(folder)?)
437            }
438            TyKind::Closure(did, args) => TyKind::Closure(did, args.try_fold_with(folder)?),
439            TyKind::CoroutineClosure(did, args) => {
440                TyKind::CoroutineClosure(did, args.try_fold_with(folder)?)
441            }
442            TyKind::Alias(kind, data) => TyKind::Alias(kind, data.try_fold_with(folder)?),
443            TyKind::Pat(ty, pat) => {
444                TyKind::Pat(ty.try_fold_with(folder)?, pat.try_fold_with(folder)?)
445            }
446
447            TyKind::Bool
448            | TyKind::Char
449            | TyKind::Str
450            | TyKind::Int(_)
451            | TyKind::Uint(_)
452            | TyKind::Float(_)
453            | TyKind::Error(_)
454            | TyKind::Infer(_)
455            | TyKind::Param(..)
456            | TyKind::Bound(..)
457            | TyKind::Placeholder(..)
458            | TyKind::Never
459            | TyKind::Foreign(..) => return Ok(self),
460        };
461
462        Ok(if self.kind() == kind { self } else { Ty::new(folder.cx(), kind) })
463    }
464    fn super_fold_with<F: rustc_type_ir::TypeFolder<DbInterner<'db>>>(
465        self,
466        folder: &mut F,
467    ) -> Self {
468        let kind = match self.kind() {
469            TyKind::RawPtr(ty, mutbl) => TyKind::RawPtr(ty.fold_with(folder), mutbl),
470            TyKind::Array(typ, sz) => TyKind::Array(typ.fold_with(folder), sz.fold_with(folder)),
471            TyKind::Slice(typ) => TyKind::Slice(typ.fold_with(folder)),
472            TyKind::Adt(tid, args) => TyKind::Adt(tid, args.fold_with(folder)),
473            TyKind::Dynamic(trait_ty, region, representation) => TyKind::Dynamic(
474                trait_ty.fold_with(folder),
475                region.fold_with(folder),
476                representation,
477            ),
478            TyKind::Tuple(ts) => TyKind::Tuple(ts.fold_with(folder)),
479            TyKind::FnDef(def_id, args) => TyKind::FnDef(def_id, args.fold_with(folder)),
480            TyKind::FnPtr(sig_tys, hdr) => TyKind::FnPtr(sig_tys.fold_with(folder), hdr),
481            TyKind::UnsafeBinder(f) => TyKind::UnsafeBinder(f.fold_with(folder)),
482            TyKind::Ref(r, ty, mutbl) => {
483                TyKind::Ref(r.fold_with(folder), ty.fold_with(folder), mutbl)
484            }
485            TyKind::Coroutine(did, args) => TyKind::Coroutine(did, args.fold_with(folder)),
486            TyKind::CoroutineWitness(did, args) => {
487                TyKind::CoroutineWitness(did, args.fold_with(folder))
488            }
489            TyKind::Closure(did, args) => TyKind::Closure(did, args.fold_with(folder)),
490            TyKind::CoroutineClosure(did, args) => {
491                TyKind::CoroutineClosure(did, args.fold_with(folder))
492            }
493            TyKind::Alias(kind, data) => TyKind::Alias(kind, data.fold_with(folder)),
494            TyKind::Pat(ty, pat) => TyKind::Pat(ty.fold_with(folder), pat.fold_with(folder)),
495
496            TyKind::Bool
497            | TyKind::Char
498            | TyKind::Str
499            | TyKind::Int(_)
500            | TyKind::Uint(_)
501            | TyKind::Float(_)
502            | TyKind::Error(_)
503            | TyKind::Infer(_)
504            | TyKind::Param(..)
505            | TyKind::Bound(..)
506            | TyKind::Placeholder(..)
507            | TyKind::Never
508            | TyKind::Foreign(..) => return self,
509        };
510
511        if self.kind() == kind { self } else { Ty::new(folder.cx(), kind) }
512    }
513}

649    fn new_coroutine_witness_for_coroutine(
650        interner: DbInterner<'db>,
651        def_id: CoroutineIdWrapper,
652        coroutine_args: <DbInterner<'db> as rustc_type_ir::Interner>::GenericArgs,
653    ) -> Self {
654        // HACK: Coroutine witness types are lifetime erased, so they
655        // never reference any lifetime args from the coroutine. We erase
656        // the regions here since we may get into situations where a
657        // coroutine is recursively contained within itself, leading to
658        // witness types that differ by region args. This means that
659        // cycle detection in fulfillment will not kick in, which leads
660        // to unnecessary overflows in async code. See the issue:
661        // <https://github.com/rust-lang/rust/issues/145151>.
662        let coroutine_args = interner.mk_args_from_iter(coroutine_args.iter().map(|arg| {
663            match arg {
664                GenericArg::Ty(_) | GenericArg::Const(_) => arg,
665                GenericArg::Lifetime(_) => {
666                    crate::next_solver::Region::new(interner, rustc_type_ir::RegionKind::ReErased)
667                        .into()
668                }
669            }
670        }));
671        Ty::new_coroutine_witness(interner, def_id, coroutine_args)
672    }

745    fn to_opt_closure_kind(self) -> Option<rustc_type_ir::ClosureKind> {
746        match self.kind() {
747            TyKind::Int(int_ty) => match int_ty {
748                IntTy::I8 => Some(ClosureKind::Fn),
749                IntTy::I16 => Some(ClosureKind::FnMut),
750                IntTy::I32 => Some(ClosureKind::FnOnce),
751                _ => unreachable!("cannot convert type `{:?}` to a closure kind", self),
752            },
753
754            // "Bound" types appear in canonical queries when the
755            // closure type is not yet known, and `Placeholder` and `Param`
756            // may be encountered in generic `AsyncFnKindHelper` goals.
757            TyKind::Bound(..) | TyKind::Placeholder(_) | TyKind::Param(_) | TyKind::Infer(_) => {
758                None
759            }
760
761            TyKind::Error(_) => Some(ClosureKind::Fn),
762
763            _ => unreachable!("cannot convert type `{:?}` to a closure kind", self),
764        }
765    }

785    fn discriminant_ty(
786        self,
787        interner: DbInterner<'db>,
788    ) -> <DbInterner<'db> as rustc_type_ir::Interner>::Ty {
789        match self.kind() {
790            TyKind::Adt(adt, _) if adt.is_enum() => adt.repr().discr_type().to_ty(interner),
791            TyKind::Coroutine(_, args) => args.as_coroutine().discr_ty(interner),
792
793            TyKind::Param(_) | TyKind::Alias(..) | TyKind::Infer(InferTy::TyVar(_)) => {
794                /*
795                let assoc_items = tcx.associated_item_def_ids(
796                    tcx.require_lang_item(hir::LangItem::DiscriminantKind, None),
797                );
798                TyKind::new_projection_from_args(tcx, assoc_items[0], tcx.mk_args(&[self.into()]))
799                */
800                unimplemented!()
801            }
802
803            TyKind::Pat(ty, _) => ty.discriminant_ty(interner),
804
805            TyKind::Bool
806            | TyKind::Char
807            | TyKind::Int(_)
808            | TyKind::Uint(_)
809            | TyKind::Float(_)
810            | TyKind::Adt(..)
811            | TyKind::Foreign(_)
812            | TyKind::Str
813            | TyKind::Array(..)
814            | TyKind::Slice(_)
815            | TyKind::RawPtr(_, _)
816            | TyKind::Ref(..)
817            | TyKind::FnDef(..)
818            | TyKind::FnPtr(..)
819            | TyKind::Dynamic(..)
820            | TyKind::Closure(..)
821            | TyKind::CoroutineClosure(..)
822            | TyKind::CoroutineWitness(..)
823            | TyKind::Never
824            | TyKind::Tuple(_)
825            | TyKind::Error(_)
826            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => {
827                Ty::new(interner, TyKind::Uint(UintTy::U8))
828            }
829
830            TyKind::Bound(..)
831            | TyKind::Placeholder(_)
832            | TyKind::Infer(
833                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
834            ) => {
835                panic!(
836                    "`dself.iter().map(|v| v.try_fold_with(folder)).collect::<Result<_, _>>()?iscriminant_ty` applied to unexpected type: {self:?}"
837                )
838            }
839            TyKind::UnsafeBinder(..) => unimplemented!(),
840        }
841    }
